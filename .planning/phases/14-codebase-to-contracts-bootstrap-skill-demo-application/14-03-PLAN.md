    ```javascript
    // scripts/discover.mjs
    import { glob } from 'tinyglobby';
    import { readFileSync } from 'node:fs';
    import { writeFileSync, mkdirSync } from 'node:fs';
    import { resolve, relative } from 'node:path';
    import { deterministicUuid } from './helpers/deterministic-uuid.mjs';
    import { callClaude } from './helpers/claude-cli-bridge.mjs';

    export async function discover(repoPath) {
      const repoName = repoPath.split('/').pop(); // 'bootstrap-demo-target'

      // 1. Enumerate by glob
      const files = {
        pages: await glob('app/**/page.tsx', { cwd: repoPath, absolute: false }),
        layouts: await glob('app/**/layout.tsx', { cwd: repoPath }),
        routes: await glob('app/api/**/route.ts', { cwd: repoPath }),
        prisma: await glob('prisma/schema.prisma', { cwd: repoPath }),
        libs: await glob('lib/**/*.ts', { cwd: repoPath }),
      };

      const nodes = [];

      // 2. Heuristic classification (per references/classification-rules.md table)
      for (const page of files.pages) {
        const route = '/' + page.replace(/^app\//, '').replace(/\/page\.tsx$/, '').replace(/^\//, '');
        const uuid = deterministicUuid(repoName, page, 'L3:default-export');
        nodes.push({
          uuid, kind: 'UI', level: 'L3', file: page,
          candidate_lines: { start_line: 1, end_line: 999 }, // refined in Stage 4
          route: route === '/' ? '/' : route,
          parent_hint: null, // populated by Stage 5a flow synthesis
        });

        // Look for L4 atoms inside the page (sectioned by H2/H3 or named exports). Defer
        // detailed AST walk to Stage 4; Stage 1 produces ROUGH candidate ranges based on
        // function-level / component-level boundaries. For each top-level JSX-returning
        // expression in the page, propose an L4 candidate.
        // (Heuristic: scan the file for `function ([A-Z]\w+)\(` and `const ([A-Z]\w+) =`
        // patterns; emit one L4 atom per pattern hit.)
      }

      for (const route of files.routes) {
        // Detect HTTP methods exported (GET / POST / DELETE / PATCH)
        const source = readFileSync(resolve(repoPath, route), 'utf8');
        const methods = ['GET', 'POST', 'PUT', 'PATCH', 'DELETE'].filter(m =>
          new RegExp(`export\\s+(async\\s+)?function\\s+${m}\\b`).test(source)
        );
        for (const method of methods) {
          const apiPath = '/' + route.replace(/^app\//, '').replace(/\/route\.ts$/, '');
          const uuid = deterministicUuid(repoName, route, `L3:${method}`);
          nodes.push({
            uuid, kind: 'API', level: 'L3', file: route,
            route: `${method} ${apiPath}`,
            parent_hint: null,
          });
        }
      }

      // Prisma models — parse schema.prisma minimally; for each `model X {` emit a data L2
      for (const schema of files.prisma) {
        const source = readFileSync(resolve(repoPath, schema), 'utf8');
        const modelRegex = /model\s+(\w+)\s*\{/g;
        let m;
        while ((m = modelRegex.exec(source)) !== null) {
          const modelName = m[1];
          const uuid = deterministicUuid(repoName, schema, `L2:model-${modelName}`);
          nodes.push({
            uuid, kind: 'data', level: 'L2', file: schema,
            model_name: modelName,
            parent_hint: null,
          });
        }
      }

      // Lib files — heuristic: import patterns determine external vs lib
      for (const lib of files.libs) {
        const source = readFileSync(resolve(repoPath, lib), 'utf8');
        const isExternal = /from ['"](stripe|@mailchimp|resend|@sendgrid|googleapis|twilio)/.test(source);
        const sizeBytes = source.length;
        const level = sizeBytes < 3000 ? 'L2' : 'L3'; // ~100-line threshold
        const kind = isExternal ? 'external' : 'lib';
        const uuid = deterministicUuid(repoName, lib, `${level}:default-export`);
        nodes.push({ uuid, kind, level, file: lib, parent_hint: null });
      }

      // 3. LLM fallback for ambiguous files
      // (Files matching multiple heuristics OR files in src/ outside lib/app/prisma)
      const ambiguous = []; // populated when heuristics conflict; minimal for v1 demo
      for (const file of ambiguous) {
        const result = await callClaude({
          schemaPath: '../schemas/classify.json',
          systemPrompt: readFileSync('./prompts/classify-atom.txt', 'utf8'),
          userPrompt: `Classify the file at ${file} (contents: ${source.slice(0, 2000)}). Return kind, level, confidence.`,
        });
        nodes.push({
          uuid: deterministicUuid(repoName, file, 'llm-classified'),
          kind: result.structured_output.kind,
          level: result.structured_output.level,
          file,
          confidence: result.structured_output.confidence,
        });
      }

      // 4. Write to .staging/
      const stagingDir = resolve(repoPath, '.contracts/.staging');
      mkdirSync(stagingDir, { recursive: true });
      writeFileSync(resolve(stagingDir, 'nodes.json'), JSON.stringify(nodes, null, 2));
      writeFileSync(resolve(stagingDir, '_progress.json'), JSON.stringify({
        stage: 1, completed_at: new Date().toISOString(), node_count: nodes.length,
      }, null, 2));

      console.log(`Stage 1 complete: ${nodes.length} candidate nodes written to .staging/nodes.json`);
      return nodes;
    }

    // CLI entry point
    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      discover(resolve(repoPath)).catch(err => { console.error(err); process.exit(1); });
    }
    ```

    **scripts/derive-frontmatter.mjs** — Stage 2 implementation:

    ```javascript
    // scripts/derive-frontmatter.mjs
    import { readFileSync, writeFileSync, existsSync, statSync } from 'node:fs';
    import { resolve } from 'node:path';
    import { createHash } from 'node:crypto';
    import { callClaude } from './helpers/claude-cli-bridge.mjs';
    import { writeFrontmatter } from './helpers/frontmatter-writer.mjs';

    export async function deriveFrontmatter(repoPath, options = {}) {
      const stagingDir = resolve(repoPath, '.contracts/.staging');
      const nodesJsonPath = resolve(stagingDir, 'nodes.json');
      if (!existsSync(nodesJsonPath)) {
        throw new Error(`Stage 1 has not been run. Expected ${nodesJsonPath}. Run discover.mjs first.`);
      }

      const nodes = JSON.parse(readFileSync(nodesJsonPath, 'utf8'));
      const systemPrompt = readFileSync(
        resolve(import.meta.url.replace('file://', '../prompts/derive-frontmatter.txt'))
      );  // resolve relative to script location

      const results = [];
      const concurrency = options.concurrency ?? 5;  // claude -p tolerates 5 concurrent

      // Hash-skip: if a frontmatter file exists and source hasn't changed, reuse it
      const eligible = nodes.filter(node => {
        const fmPath = resolve(stagingDir, `${node.uuid}.frontmatter.json`);
        const sourcePath = resolve(repoPath, node.file);
        if (!existsSync(sourcePath)) return false; // file deleted; skip
        if (!existsSync(fmPath)) return true; // no prior derivation; need to run

        // Compare source-content sha256 to stored hash in the existing frontmatter
        const sourceHash = createHash('sha256').update(readFileSync(sourcePath)).digest('hex');
        const existing = JSON.parse(readFileSync(fmPath, 'utf8'));
        return existing._source_sha256 !== sourceHash;
      });

      console.log(`Stage 2: ${eligible.length}/${nodes.length} nodes need (re-)derivation (${nodes.length - eligible.length} hash-skipped)`);

      // Parallel batched processing
      for (let i = 0; i < eligible.length; i += concurrency) {
        const batch = eligible.slice(i, i + concurrency);
        await Promise.all(batch.map(async node => {
          const sourcePath = resolve(repoPath, node.file);
          const sourceContent = readFileSync(sourcePath, 'utf8');
          const sourceHash = createHash('sha256').update(sourceContent).digest('hex');

          const userPrompt = `Generate frontmatter for this candidate node.
Candidate: ${JSON.stringify(node, null, 2)}
File path (relative): ${node.file}
Source contents (first 5000 chars):
\`\`\`typescript
${sourceContent.slice(0, 5000)}
\`\`\`
Output the frontmatter as structured JSON per the schema.`;

          let result;
          try {
            result = await callClaude({
              schemaPath: resolve(import.meta.url.replace('file://', '../schemas/frontmatter.json')),
              systemPrompt,
              userPrompt,
              temperature: 0,
              allowedTools: ['Read'],
            });
          } catch (err) {
            console.error(`Failed to derive frontmatter for ${node.uuid} (${node.file}): ${err.message}`);
            return;
          }

          const fm = result.structured_output;
          // Persist + tag with source hash for next-run hash-skip
          fm._source_sha256 = sourceHash;
          writeFileSync(
            resolve(stagingDir, `${node.uuid}.frontmatter.json`),
            JSON.stringify(fm, null, 2)
          );
          results.push({ uuid: node.uuid, status: 'derived' });
        }));

        console.log(`  Stage 2 progress: ${Math.min(i + concurrency, eligible.length)}/${eligible.length}`);
      }

      // Update _progress.json
      const progressPath = resolve(stagingDir, '_progress.json');
      const progress = JSON.parse(readFileSync(progressPath, 'utf8'));
      progress.stage = 2;
      progress.stage_2_completed_at = new Date().toISOString();
      progress.stage_2_derived_count = results.length;
      progress.stage_2_skipped_count = nodes.length - eligible.length;
      writeFileSync(progressPath, JSON.stringify(progress, null, 2));

      console.log(`Stage 2 complete: ${results.length} frontmatter files written, ${nodes.length - eligible.length} hash-skipped`);
      return results;
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      deriveFrontmatter(resolve(repoPath)).catch(err => { console.error(err); process.exit(1); });
    }
    ```

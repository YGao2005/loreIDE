    ```javascript
    // scripts/derive-body.mjs
    import { readFileSync, writeFileSync, existsSync, readdirSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import { callClaude } from './helpers/claude-cli-bridge.mjs';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = dirname(__filename);

    export async function deriveBody(repoPath, options = {}) {
      const stagingDir = resolve(repoPath, '.contracts/.staging');
      const nodes = JSON.parse(readFileSync(resolve(stagingDir, 'nodes.json'), 'utf8'));
      const systemPrompt = readFileSync(resolve(__dirname, '../prompts/derive-body.txt'), 'utf8');

      const concurrency = options.concurrency ?? 5;
      const results = [];

      // Group nodes by L3 surface for context coherence — page.tsx + its child atoms together
      const byParent = new Map();
      for (const node of nodes) {
        const parentKey = node.parent_hint || node.uuid; // fallback to self for L3 roots
        if (!byParent.has(parentKey)) byParent.set(parentKey, []);
        byParent.get(parentKey).push(node);
      }

      // Process each L3 group sequentially (cache hits within a group); group-level parallelism
      const groups = [...byParent.values()];
      for (let i = 0; i < groups.length; i += concurrency) {
        const batch = groups.slice(i, i + concurrency);
        await Promise.all(batch.map(async group => {
          // For each node in group, derive body
          for (const node of group) {
            const bodyPath = resolve(stagingDir, `${node.uuid}.body.json`);
            const sourcePath = resolve(repoPath, node.file);
            if (!existsSync(sourcePath)) continue;

            // Hash-skip: skip if body.json exists and is fresh
            if (existsSync(bodyPath)) {
              // (Optional: hash-check; for v1, skip if file present per progress.json)
              const progress = JSON.parse(readFileSync(resolve(stagingDir, '_progress.json'), 'utf8'));
              if (progress.stage_3_completed_for?.includes(node.uuid)) continue;
            }

            const sourceContent = readFileSync(sourcePath, 'utf8');
            const userPrompt = `Derive the contract body for this node.
Node: ${JSON.stringify(node, null, 2)}
File: ${node.file}
Source contents (first 5000 chars):
\`\`\`typescript
${sourceContent.slice(0, 5000)}
\`\`\`
Output structured JSON per the schema with intent, role, and (for UI L4) examples or (for backend kinds) inputs / outputs / side_effects.`;

            try {
              const result = await callClaude({
                schemaPath: resolve(__dirname, '../schemas/contract-body.json'),
                systemPrompt,
                userPrompt,
                temperature: 0,
                allowedTools: ['Read'],
              });
              writeFileSync(bodyPath, JSON.stringify(result.structured_output, null, 2));
              results.push({ uuid: node.uuid, status: 'derived' });
            } catch (err) {
              console.error(`Failed to derive body for ${node.uuid} (${node.file}): ${err.message}`);
            }
          }
        }));
        console.log(`  Stage 3 progress: ${Math.min(i + concurrency, groups.length)}/${groups.length} L3 groups`);
      }

      // Update progress
      const progressPath = resolve(stagingDir, '_progress.json');
      const progress = JSON.parse(readFileSync(progressPath, 'utf8'));
      progress.stage = 3;
      progress.stage_3_completed_at = new Date().toISOString();
      progress.stage_3_derived_count = results.length;
      writeFileSync(progressPath, JSON.stringify(progress, null, 2));

      console.log(`Stage 3 complete: ${results.length} body files written to .staging/`);
      return results;
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      deriveBody(resolve(repoPath)).catch(err => { console.error(err); process.exit(1); });
    }
    ```

    **scripts/align-jsx.mjs** — Stage 4 implementation per RESEARCH Pattern 4:

    ```javascript
    // scripts/align-jsx.mjs
    import { readFileSync, writeFileSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import { loadBabel } from './helpers/babel-parser-bridge.mjs';
    import { callClaude } from './helpers/claude-cli-bridge.mjs';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = dirname(__filename);

    /**
     * Align an L4 UI atom candidate to its outermost JSX element.
     * Returns { code_ranges: [...] } on success;
     * { error: 'no_match' | 'needs_tiebreak', candidates: [...] } on ambiguity / miss.
     *
     * MUST use the same @babel/parser config as Phase 9 webpack loader:
     *   sourceType: 'module', plugins: ['jsx', 'typescript']
     * Drift here breaks validator parity with jsx_align_validator.rs (Pitfall 7).
     */
    export async function alignAtomToJsx({ atomNode, parentSource, babel }) {
      const ast = babel.parse(parentSource, {
        sourceType: 'module',
        plugins: ['jsx', 'typescript'],
        errorRecovery: true,  // tolerate minor parse errors; validator is the final gate
      });

      const matches = [];
      const candidateStart = atomNode.candidate_lines?.start_line ?? 1;
      const candidateEnd = atomNode.candidate_lines?.end_line ?? 999;

      babel.traverse(ast, {
        JSXElement(path) {
          const start = path.node.loc?.start?.line;
          const end = path.node.loc?.end?.line;
          if (start == null || end == null) return;

          // Match if FULLY contained in candidate range
          if (start >= candidateStart && end <= candidateEnd) {
            // Outermost check: no ancestor JSXElement also fully contained
            let isOutermost = true;
            let parent = path.parentPath;
            while (parent) {
              if (parent.node?.type === 'JSXElement') {
                const ps = parent.node.loc?.start?.line;
                const pe = parent.node.loc?.end?.line;
                if (ps != null && pe != null && ps >= candidateStart && pe <= candidateEnd) {
                  isOutermost = false;
                  break;
                }
              }
              parent = parent.parentPath;
            }
            if (isOutermost) {
              matches.push({
                start_line: start,
                end_line: end,
                snippet: parentSource.split('\n').slice(start - 1, end).join('\n'),
              });
            }
          }
        },
      });

      if (matches.length === 0) return { error: 'no_match', atomNode };
      if (matches.length === 1) {
        return {
          code_ranges: [{ file: atomNode.file, start_line: matches[0].start_line, end_line: matches[0].end_line }],
        };
      }

      // Multi-match: LLM tiebreak — pass intent + candidate snippets, ask claude to pick top-1
      const tiebreakPrompt = `An L4 UI atom has multiple candidate JSX elements in its parent file.
Atom intent: "${atomNode.intent_summary || atomNode.uuid}"
File: ${atomNode.file}
Candidates (numbered):
${matches.map((m, i) => `${i + 1}. Lines ${m.start_line}-${m.end_line}:\n\`\`\`tsx\n${m.snippet.slice(0, 500)}\n\`\`\``).join('\n\n')}

Respond with ONLY the index (1-based) of the candidate that best matches the atom's intent.`;

      const tiebreakResult = await callClaude({
        systemPrompt: 'You are picking the JSX element that best matches a given intent. Respond with a single integer (the candidate index).',
        userPrompt: tiebreakPrompt,
        temperature: 0,
        allowedTools: [],
      });

      // Parse the integer response (no schema for this single-token output)
      const pickIndex = parseInt(String(tiebreakResult.raw).trim().match(/\d+/)?.[0] || '1', 10) - 1;
      const picked = matches[Math.max(0, Math.min(pickIndex, matches.length - 1))];

      return {
        code_ranges: [{ file: atomNode.file, start_line: picked.start_line, end_line: picked.end_line }],
        tiebreak: true,
      };
    }

    export async function alignAllAtoms(repoPath) {
      const stagingDir = resolve(repoPath, '.contracts/.staging');
      const nodes = JSON.parse(readFileSync(resolve(stagingDir, 'nodes.json'), 'utf8'));
      const babel = await loadBabel(repoPath);

      const failures = [];
      const updated = [];

      // Filter to L4 UI atoms only — backend kinds skip (JSX-01 exempt per BACKEND-FM-01)
      const l4UiAtoms = nodes.filter(n => n.level === 'L4' && n.kind === 'UI');

      // Group by parent file (one .tsx per group; parse once per file, traverse N times)
      const byFile = new Map();
      for (const atom of l4UiAtoms) {
        if (!byFile.has(atom.file)) byFile.set(atom.file, []);
        byFile.get(atom.file).push(atom);
      }

      for (const [file, atoms] of byFile.entries()) {
        const sourcePath = resolve(repoPath, file);
        const source = readFileSync(sourcePath, 'utf8');

        for (const atom of atoms) {
          const result = await alignAtomToJsx({ atomNode: atom, parentSource: source, babel });
          if (result.error === 'no_match') {
            failures.push({ uuid: atom.uuid, file: atom.file, reason: 'no JSX element matches candidate range' });
            continue;
          }
          // Update the atom's frontmatter file in .staging/
          const fmPath = resolve(stagingDir, `${atom.uuid}.frontmatter.json`);
          const fm = JSON.parse(readFileSync(fmPath, 'utf8'));
          fm.code_ranges = result.code_ranges;
          writeFileSync(fmPath, JSON.stringify(fm, null, 2));
          updated.push({ uuid: atom.uuid, code_ranges: result.code_ranges, tiebreak: !!result.tiebreak });
        }
      }

      // CRITICAL: per RESEARCH Pattern 4 + Pitfall 7, refuse to proceed if any L4 UI atom failed
      if (failures.length > 0) {
        const failuresPath = resolve(stagingDir, '_stage4_failures.json');
        writeFileSync(failuresPath, JSON.stringify(failures, null, 2));
        console.error(`Stage 4 FAILED: ${failures.length} L4 UI atom(s) could not be aligned to any JSX element.`);
        console.error(`Details: ${failuresPath}`);
        console.error(`Refusing to proceed to Stage 5 — fix the candidate ranges or accept these atoms as unbootstrappable.`);
        process.exit(1);
      }

      // Update _progress.json
      const progressPath = resolve(stagingDir, '_progress.json');
      const progress = JSON.parse(readFileSync(progressPath, 'utf8'));
      progress.stage = 4;
      progress.stage_4_completed_at = new Date().toISOString();
      progress.stage_4_aligned_count = updated.length;
      progress.stage_4_tiebreak_count = updated.filter(u => u.tiebreak).length;
      writeFileSync(progressPath, JSON.stringify(progress, null, 2));

      console.log(`Stage 4 complete: ${updated.length} L4 UI atoms aligned (${updated.filter(u => u.tiebreak).length} via LLM tiebreak)`);
      return updated;
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      alignAllAtoms(resolve(repoPath)).catch(err => { console.error(err); process.exit(1); });
    }
    ```

    // scripts/synthesize-flows.mjs
    import { readFileSync, writeFileSync, existsSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import { loadBabel } from './helpers/babel-parser-bridge.mjs';
    import { callClaude } from './helpers/claude-cli-bridge.mjs';
    import { deterministicUuid } from './helpers/deterministic-uuid.mjs';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = dirname(__filename);

    /**
     * Walk import statements in `triggerSource` to find files that exist in our nodes.json,
     * then walk the AST for invocation order.
     * Returns array of { uuid, file, name, line } in invocation order.
     */
    export function staticCallChain({ triggerSource, triggerFile, allNodes, babel }) {
      const ast = babel.parse(triggerSource, {
        sourceType: 'module',
        plugins: ['jsx', 'typescript'],
        errorRecovery: true,
      });

      // 1. Collect imports — `import { fn } from './lib/x'`
      const importMap = new Map(); // local-name → resolved-file
      babel.traverse(ast, {
        ImportDeclaration(path) {
          const source = path.node.source.value;
          if (!source.startsWith('.') && !source.startsWith('@/')) return; // skip third-party
          // Resolve relative to triggerFile
          const triggerDir = dirname(triggerFile);
          let resolvedFile = source.startsWith('@/')
            ? source.replace('@/', 'src/')
            : resolve(triggerDir, source);
          if (!resolvedFile.endsWith('.ts') && !resolvedFile.endsWith('.tsx')) {
            // Try .ts then .tsx then index.ts
            for (const ext of ['.ts', '.tsx', '/index.ts']) {
              if (allNodes.find(n => n.file === resolvedFile + ext || n.file === resolvedFile.replace(/\\$/, '') + ext)) {
                resolvedFile = resolvedFile + ext;
                break;
              }
            }
          }
          for (const spec of path.node.specifiers || []) {
            const localName = spec.local.name;
            const node = allNodes.find(n => n.file === resolvedFile);
            if (node) importMap.set(localName, { file: resolvedFile, uuid: node.uuid });
          }
        },
      });

      // 2. Walk for call-sites in invocation order — preorder traversal
      const calls = [];
      babel.traverse(ast, {
        CallExpression(path) {
          // Identify callee name
          let calleeName;
          if (path.node.callee.type === 'Identifier') calleeName = path.node.callee.name;
          else if (path.node.callee.type === 'MemberExpression' && path.node.callee.object.type === 'Identifier') {
            // Match `db.user.update`-style: object becomes the import name
            calleeName = path.node.callee.object.name;
          }
          if (!calleeName) return;
          const importInfo = importMap.get(calleeName);
          if (importInfo) {
            calls.push({
              uuid: importInfo.uuid,
              file: importInfo.file,
              name: calleeName,
              line: path.node.loc?.start?.line ?? 0,
            });
          }
        },
      });

      // Dedup but preserve first-call order (Stripe.customers.update called twice → one member)
      const seen = new Set();
      return calls.filter(c => {
        if (seen.has(c.uuid)) return false;
        seen.add(c.uuid);
        return true;
      });
    }

    export async function synthesizeFlows(repoPath, options = {}) {
      const stagingDir = resolve(repoPath, '.contracts/.staging');
      const nodes = JSON.parse(readFileSync(resolve(stagingDir, 'nodes.json'), 'utf8'));
      const babel = await loadBabel(repoPath);

      // Triggers = L3 surfaces (UI page or API route)
      const triggers = nodes.filter(n => n.level === 'L3' && (n.kind === 'UI' || n.kind === 'API'));

      const flows = [];
      for (const trigger of triggers) {
        const sourcePath = resolve(repoPath, trigger.file);
        if (!existsSync(sourcePath)) continue;
        const triggerSource = readFileSync(sourcePath, 'utf8');

        // 1. Static AST walk
        const staticChain = staticCallChain({ triggerSource, triggerFile: trigger.file, allNodes: nodes, babel });

        // Recurse one level: for each member, walk ITS imports + calls and append unseen
        const seen = new Set([trigger.uuid, ...staticChain.map(c => c.uuid)]);
        for (const member of staticChain) {
          const memberPath = resolve(repoPath, member.file);
          if (!existsSync(memberPath)) continue;
          const memberSource = readFileSync(memberPath, 'utf8');
          const subChain = staticCallChain({ triggerSource: memberSource, triggerFile: member.file, allNodes: nodes, babel });
          for (const sub of subChain) {
            if (!seen.has(sub.uuid)) {
              staticChain.push(sub);
              seen.add(sub.uuid);
            }
          }
        }

        if (staticChain.length === 0) continue; // trigger has no participants — not a flow worth modeling

        // 2. LLM verification (single call per flow)
        const proposedMembers = [trigger.uuid, ...staticChain.map(c => c.uuid)];
        const userPrompt = `Verify this flow's members: ordering.
Trigger: ${trigger.uuid} (${trigger.file}, kind: ${trigger.kind})
Trigger source:
\`\`\`typescript
${triggerSource.slice(0, 3000)}
\`\`\`
Proposed members in order:
${proposedMembers.map((uuid, i) => {
  const node = nodes.find(n => n.uuid === uuid);
  return `${i + 1}. ${uuid} — ${node?.file} (${node?.kind})`;
}).join('\n')}

Reorder, add, or remove members to reflect ACTUAL invocation order. Output as JSON per the flow schema.`;

        let verifiedMembers = proposedMembers;
        try {
          const result = await callClaude({
            schemaPath: resolve(__dirname, '../schemas/flow.json'),
            systemPrompt: readFileSync(resolve(__dirname, '../prompts/synthesize-flow.txt'), 'utf8'),
            userPrompt,
            temperature: 0,
            allowedTools: ['Read'],
          });
          verifiedMembers = result.structured_output.members;
        } catch (err) {
          console.warn(`Flow synthesis LLM verification failed for trigger ${trigger.uuid}: ${err.message}. Using static chain.`);
        }

        // 3. Emit flow contract
        const flowSlug = trigger.file.replace(/[^\w]/g, '-').replace(/^-+|-+$/g, '').slice(0, 50);
        const flowUuid = deterministicUuid(repoPath.split('/').pop(), `flow-${flowSlug}`, 'L2:flow');
        const flowFm = {
          format_version: 5,
          uuid: flowUuid,
          kind: 'flow',
          level: 'L2',
          parent: null,
          neighbors: [],
          members: verifiedMembers,
          code_ranges: [],
          code_hash: null,
          contract_hash: null,
          human_pinned: false,
          route: null,
          derived_at: null,
          section_hashes: {},
          rollup_inputs: [],
          rollup_hash: null,
          rollup_state: 'untracked',
          rollup_generation: 0,
        };
        const flowBody = {
          intent: `Auto-derived flow originating at ${trigger.file}. Walks ${verifiedMembers.length - 1} downstream participants. Body to be refined by IDE on first PROP-02 rollup recompute.`,
          role: `Bootstrapped flow contract — synthesized from import chain + AST call-sites.`,
        };

        writeFileSync(
          resolve(stagingDir, `${flowUuid}.frontmatter.json`),
          JSON.stringify(flowFm, null, 2)
        );
        writeFileSync(
          resolve(stagingDir, `${flowUuid}.body.json`),
          JSON.stringify(flowBody, null, 2)
        );

        flows.push({ uuid: flowUuid, trigger: trigger.uuid, members: verifiedMembers });
      }

      // Update progress
      const progressPath = resolve(stagingDir, '_progress.json');
      const progress = JSON.parse(readFileSync(progressPath, 'utf8'));
      progress.stage = 5;
      progress.stage_5a_completed_at = new Date().toISOString();
      progress.stage_5a_flows_synthesized = flows.length;
      writeFileSync(progressPath, JSON.stringify(progress, null, 2));

      console.log(`Stage 5a complete: ${flows.length} flow contracts synthesized`);
      return flows;
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      synthesizeFlows(resolve(repoPath)).catch(err => { console.error(err); process.exit(1); });
    }

    // scripts/install-babel-plugin.mjs
    import { existsSync, readFileSync, writeFileSync, mkdirSync, copyFileSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = dirname(__filename);

    /**
     * Copy contract-uuid-plugin/ scaffold into target repo.
     * Idempotent — re-running replaces files (no append).
     */
    export function installBabelPlugin(repoPath) {
      const pluginDir = resolve(repoPath, 'contract-uuid-plugin');
      mkdirSync(pluginDir, { recursive: true });
      copyFileSync(
        resolve(__dirname, '../templates/contract-uuid-plugin-loader.js'),
        resolve(pluginDir, 'index.js')
      );
      copyFileSync(
        resolve(__dirname, '../templates/contract-uuid-plugin-package.json'),
        resolve(pluginDir, 'package.json')
      );

      // Patch next.config.{ts,js,mjs}
      const candidates = ['next.config.ts', 'next.config.js', 'next.config.mjs'];
      const configFile = candidates.map(c => resolve(repoPath, c)).find(p => existsSync(p));
      if (!configFile) {
        console.warn('No next.config.{ts,js,mjs} found in target repo. Plugin scaffold installed; user must wire manually.');
        return { plugin: 'installed', configPatched: false };
      }

      const snippetPath = resolve(__dirname, '../templates/next-config-snippet.ts');
      const snippet = readFileSync(snippetPath, 'utf8');
      let config = readFileSync(configFile, 'utf8');

      // Idempotent patch — find existing BOOTSTRAP-INSERT block and replace; else inject before module.exports
      const startMarker = '// BOOTSTRAP-INSERT-START';
      const endMarker = '// BOOTSTRAP-INSERT-END';
      const startIdx = config.indexOf(startMarker);
      const endIdx = config.indexOf(endMarker);

      if (startIdx >= 0 && endIdx >= 0) {
        config = config.slice(0, startIdx) + snippet + config.slice(endIdx + endMarker.length);
      } else {
        // Insert before "export default" / "module.exports"
        const exportMatch = config.match(/^(export\s+default|module\.exports)/m);
        if (exportMatch) {
          const insertAt = config.indexOf(exportMatch[0]);
          config = config.slice(0, insertAt) + '\n' + snippet + '\n' + config.slice(insertAt);
        } else {
          config = config + '\n' + snippet + '\n';
        }
      }

      writeFileSync(configFile, config);
      return { plugin: 'installed', configPatched: configFile };
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      const result = installBabelPlugin(resolve(repoPath));
      console.log(JSON.stringify(result, null, 2));
    }

    // scripts/validate.mjs
    import { execSync, spawnSync } from 'node:child_process';
    import { existsSync, readFileSync, readdirSync } from 'node:fs';
    import { resolve } from 'node:path';

    /**
     * Validator gate.
     * Strategy: try `contract-ide validate-repo <path>` subprocess (source of truth).
     * Falls back to JS-side reimplementation with a warning.
     */
    export function validate(repoPath, options = {}) {
      // 1. Try IDE binary
      const ideBinaryAvailable = (() => {
        try {
          execSync('which contract-ide', { stdio: 'pipe' });
          return true;
        } catch { return false; }
      })();

      if (ideBinaryAvailable && !options.forceJsValidator) {
        try {
          const result = spawnSync('contract-ide', ['validate-repo', repoPath], { encoding: 'utf8' });
          if (result.status === 0) return { ok: true, source: 'ide-binary' };
          return {
            ok: false,
            source: 'ide-binary',
            stderr: result.stderr,
            errors: result.stderr.split('\n').filter(Boolean),
          };
        } catch (err) {
          console.warn('contract-ide validate-repo failed; falling back to JS validator');
        }
      }

      // 2. JS-side fallback — reimplement JSX-01 + BACKEND-FM-01 minimally
      console.warn('Skill is using degraded JS-side validators. Install Contract IDE for stronger guarantees (or add `validate-repo` CLI subcommand to the binary).');

      const stagingDir = resolve(repoPath, '.contracts/.staging');
      const errors = [];

      // Read all sidecars in .staging/ (or .contracts/ if staging promoted already)
      const sourceDir = existsSync(stagingDir) ? stagingDir : resolve(repoPath, '.contracts');
      if (!existsSync(sourceDir)) return { ok: false, errors: [`No .staging/ or .contracts/ directory at ${sourceDir}`] };

      const sidecars = readdirSync(sourceDir).filter(f => f.endsWith('.md'));
      for (const file of sidecars) {
        const content = readFileSync(resolve(sourceDir, file), 'utf8');
        const fmMatch = content.match(/^---\n([\s\S]*?)\n---/);
        if (!fmMatch) {
          errors.push(`${file}: no YAML frontmatter`);
          continue;
        }

        // BACKEND-FM-01 fallback: backend kinds need ## Inputs / ## Outputs / ## Side effects sections
        const isBackend = /\bkind:\s*(API|lib|data|external|job|cron|event)\b/.test(fmMatch[1]);
        if (isBackend) {
          if (!/^## Inputs\b/m.test(content)) errors.push(`${file}: missing ## Inputs (BACKEND-FM-01)`);
          if (!/^## Outputs\b/m.test(content)) errors.push(`${file}: missing ## Outputs (BACKEND-FM-01)`);
          if (!/^## Side effects\b/m.test(content)) errors.push(`${file}: missing ## Side effects (BACKEND-FM-01)`);
        }

        // JSX-01 fallback: L4 UI atoms must have non-empty code_ranges with ≥1 entry
        const isL4Ui = /\bkind:\s*UI\b/.test(fmMatch[1]) && /\blevel:\s*L4\b/.test(fmMatch[1]);
        if (isL4Ui) {
          const cr = fmMatch[1].match(/code_ranges:\s*\n((?:\s*-\s*.+\n?)+)/);
          if (!cr || cr[1].trim() === '') errors.push(`${file}: L4 UI atom has empty code_ranges (JSX-01)`);
        }
      }

      return { ok: errors.length === 0, source: 'js-fallback', errors };
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      const result = validate(resolve(repoPath));
      if (!result.ok) {
        console.error('Validation FAILED:');
        for (const err of result.errors) console.error(`  - ${err}`);
        process.exit(1);
      }
      console.log(`Validation OK (source: ${result.source})`);
    }

    import { existsSync, readFileSync, writeFileSync, readdirSync, renameSync, mkdirSync, rmSync } from 'node:fs';
    import { resolve, dirname } from 'node:path';
    import { fileURLToPath } from 'node:url';
    import { writeFrontmatter } from './helpers/frontmatter-writer.mjs';
    import { installBabelPlugin } from './install-babel-plugin.mjs';
    import { validate } from './validate.mjs';

    const __filename = fileURLToPath(import.meta.url);
    const __dirname = dirname(__filename);

    export async function emit(repoPath) {
      const stagingDir = resolve(repoPath, '.contracts/.staging');
      const finalDir = resolve(repoPath, '.contracts');

      if (!existsSync(stagingDir)) throw new Error(`Staging directory missing: ${stagingDir}. Run earlier stages first.`);

      // 1. Compose .md sidecars in .staging/<uuid>.md from frontmatter.json + body.json
      const fmFiles = readdirSync(stagingDir).filter(f => f.endsWith('.frontmatter.json'));
      let sidecarsWritten = 0;
      for (const fmFile of fmFiles) {
        const uuid = fmFile.replace('.frontmatter.json', '');
        const fm = JSON.parse(readFileSync(resolve(stagingDir, fmFile), 'utf8'));
        delete fm._source_sha256; // strip staging-only marker

        const bodyPath = resolve(stagingDir, `${uuid}.body.json`);
        const body = existsSync(bodyPath) ? JSON.parse(readFileSync(bodyPath, 'utf8')) : { intent: '(body missing)', role: '' };

        // Compose markdown body from sections
        let mdBody = `## Intent\n${body.intent || ''}\n\n## Role\n${body.role || ''}`;
        if (body.examples) {
          mdBody += '\n\n## Examples\n' + (body.examples.length ? body.examples.join('\n\n') : '(none yet)');
        }
        if (body.inputs) {
          mdBody += '\n\n## Inputs\n' + body.inputs.map(i => `- ${i}`).join('\n');
        }
        if (body.outputs) {
          mdBody += '\n\n## Outputs\n' + body.outputs.map(o => `- ${o}`).join('\n');
        }
        if (body.side_effects !== undefined) {
          mdBody += '\n\n## Side effects\n' + (body.side_effects.length ? body.side_effects.map(s => `- ${s}`).join('\n') : '(none)');
        }
        if (body.notes) {
          mdBody += '\n\n## Notes\n' + body.notes;
        }

        const sidecarPath = resolve(stagingDir, `${uuid}.md`);
        writeFileSync(sidecarPath, writeFrontmatter(fm, mdBody));
        sidecarsWritten++;
      }
      console.log(`Composed ${sidecarsWritten} sidecars in .staging/`);

      // 2. Install Babel plugin scaffold (idempotent)
      const pluginResult = installBabelPlugin(repoPath);
      console.log(`Babel plugin: ${JSON.stringify(pluginResult)}`);

      // 3. Validator gate — run BEFORE atomic rename
      const validation = validate(repoPath);
      if (!validation.ok) {
        console.error('Validation failed; .staging/ NOT promoted to .contracts/');
        for (const err of validation.errors) console.error(`  - ${err}`);
        process.exit(1);
      }
      console.log(`Validation OK (source: ${validation.source})`);

      // 4. Atomic rename — .staging/ → .contracts/
      // First, stash .json staging artifacts elsewhere (don't promote to .contracts/)
      const archiveDir = resolve(repoPath, '.contracts/.bootstrap-staging-archive');
      mkdirSync(archiveDir, { recursive: true });
      for (const file of readdirSync(stagingDir)) {
        if (file.endsWith('.json')) {
          renameSync(resolve(stagingDir, file), resolve(archiveDir, file));
        }
      }

      // Now promote .staging/ contents (only .md sidecars + flow .md) up
      if (existsSync(finalDir) && finalDir !== stagingDir) {
        // Backup existing .contracts/ if present; for v1 demo, error if it has files
        const existingFiles = readdirSync(finalDir).filter(f => !f.startsWith('.') && f !== 'ambient');
        if (existingFiles.length > 0) {
          console.error(`.contracts/ already has ${existingFiles.length} sidecars. Aborting to avoid clobber. Move .contracts/ aside and retry.`);
          process.exit(1);
        }
      }
      mkdirSync(finalDir, { recursive: true });
      for (const file of readdirSync(stagingDir)) {
        renameSync(resolve(stagingDir, file), resolve(finalDir, file));
      }

      // Cleanup staging dir (now empty)
      try { rmSync(stagingDir, { recursive: true, force: true }); } catch {}

      console.log(`Stage 5b complete: ${sidecarsWritten} sidecars + plugin installed in ${repoPath}/.contracts/`);
      return { sidecarsWritten, pluginInstalled: true, validatorSource: validation.source };
    }

    if (import.meta.url === `file://${process.argv[1]}`) {
      const repoPath = process.argv[2] || process.cwd();
      emit(resolve(repoPath)).catch(err => { console.error(err); process.exit(1); });
    }

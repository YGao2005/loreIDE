       ```ts
       import type { MassMatchResult } from '@/ipc/mass-edit';

       interface MassEditStore {
         /** From 09-01 — uuid → animationDelayMs for the staggered pulse */
         matchedUuids: Map<string, number>;
         setMatches: (uuids: string[]) => void;
         clearMatches: () => void;

         /** NEW in 09-02 — review queue state */

         /** Mirrors response.embedding_status from findByIntentMass; UI uses this to render the keyword-only notice */

         /** Snapshot of rollupStaleUuids size BEFORE the apply runs — diff against post-apply gives the cascade count */
         rollupStaleAtStart: number;

         setQuery: (q: string) => void;
         setCandidates: (c: MassMatchResult[]) => void;
         setEmbeddingStatus: (s: 'disabled' | 'enabled') => void;
         toggleSelected: (uuid: string) => void;
         setAllSelected: (uuids: string[]) => void;
         setApplyState: (s: MassEditStore['applyState']) => void;
         setResult: (r: MassEditStore['result']) => void;
         resetReviewQueue: () => void;
       }

       export const useMassEditStore = create<MassEditStore>((set, get) => ({
         matchedUuids: new Map(),
         setMatches: (uuids) => set({ matchedUuids: new Map(uuids.map((u, i) => [u, i * 50])) }),
         clearMatches: () => set({ matchedUuids: new Map() }),

         query: '',
         candidates: [],
         embeddingStatus: 'disabled',  // 09-01 ships keyword-only fallback per MASS-01 spec
         selectedUuids: new Set(),
         applyState: 'idle',
         result: null,
         rollupStaleAtStart: 0,

         setQuery: (q) => set({ query: q }),
         setCandidates: (c) => set({
           candidates: c,
           selectedUuids: new Set(c.filter(n => !isPinned(n)).map(n => n.uuid)), // pinned default-deselected
         }),
         setEmbeddingStatus: (s) => set({ embeddingStatus: s }),
         toggleSelected: (uuid) => {
           const next = new Set(get().selectedUuids);
           if (next.has(uuid)) next.delete(uuid); else next.add(uuid);
           set({ selectedUuids: next });
         },
         setAllSelected: (uuids) => set({ selectedUuids: new Set(uuids) }),
         setApplyState: (s) => set({ applyState: s }),
         setResult: (r) => set({ result: r }),
         resetReviewQueue: () => set({
           query: '', candidates: [], embeddingStatus: 'disabled',
           selectedUuids: new Set(), applyState: 'idle', result: null, rollupStaleAtStart: 0,
         }),
       }));

       function isPinned(n: MassMatchResult): boolean {
         // 09-01 surfaces human_pinned: boolean as a first-class field on every MassMatchResult.
         // No fallback — if the field isn't present, that's a 09-01 regression to fix in 09-01,
         // not absorb here. The dep direction is strictly 09-01 → 09-02 (no back-edits).
         return n.human_pinned === true;
       }
       ```

       ```tsx
       import { useState } from 'react';
       import { useMassEditStore } from '@/store/massEdit';
       import { useRollupStore } from '@/store/rollup'; // Phase 8 PROP-02
       import { findByIntentMass } from '@/ipc/mass-edit';
       import { MassEditModal } from './MassEditModal';
       import { Dialog, DialogContent, DialogHeader, DialogTitle } from '@/components/ui/dialog';
       import { Input } from '@/components/ui/input';
       import { Button } from '@/components/ui/button';

       interface Props { open: boolean; onClose: () => void; }
       type Stage = 'query' | 'pulsing' | 'modal' | 'no-results';

       export function MassEditTrigger({ open, onClose }: Props) {
         const [stage, setStage] = useState<Stage>('query');
         const [draft, setDraft] = useState('');
         const setQuery = useMassEditStore(s => s.setQuery);
         const setCandidates = useMassEditStore(s => s.setCandidates);
         const setEmbeddingStatus = useMassEditStore(s => s.setEmbeddingStatus);
         const setMatches = useMassEditStore(s => s.setMatches);
         const clearMatches = useMassEditStore(s => s.clearMatches);
         const resetReviewQueue = useMassEditStore(s => s.resetReviewQueue);

         async function handleSubmit() {
           setQuery(draft);
           const response = await findByIntentMass(draft, 100);
           if (response.matches.length === 0) {
             setStage('no-results');
             return;
           }
           setCandidates(response.matches);
           // Capture embedding_status so MassEditModal can render the keyword-only notice
           setEmbeddingStatus(response.embedding_status);
           setMatches(response.matches.map(m => m.uuid));
           // Snapshot rollup stale BEFORE apply so we can diff later
           useMassEditStore.setState({ rollupStaleAtStart: useRollupStore.getState().rollupStaleUuids.size });
           setStage('pulsing');
           // Hold the pulse for 3s before opening modal — must_haves truth #2
           setTimeout(() => setStage('modal'), 3000);
         }

         function handleClose() {
           clearMatches();
           resetReviewQueue();
           setStage('query');
           setDraft('');
           onClose();
         }

         if (stage === 'modal') {
           return <MassEditModal open onClose={handleClose} />;
         }

         return (
           <Dialog open={open} onOpenChange={(o) => !o && handleClose()}>
             <DialogContent>
               {stage === 'query' && (<>
                 <DialogHeader><DialogTitle>Mass edit by intent</DialogTitle></DialogHeader>
                 <Input value={draft} onChange={e => setDraft(e.target.value)}
                        placeholder="add audit logging to every destructive endpoint" />
                 <Button onClick={handleSubmit}>Find matches</Button>
               </>)}
               {stage === 'pulsing' && (<div className="p-6 text-center">
                 <p className="text-sm">{useMassEditStore.getState().candidates.length} nodes match — reviewing in 3s…</p>
               </div>)}
               {stage === 'no-results' && (<div className="p-6">
                 <p>No matches for "{draft}".</p>
                 <Button onClick={() => setStage('query')}>Try another</Button>
                 <Button variant="ghost" onClick={handleClose}>Cancel</Button>
               </div>)}
             </DialogContent>
           </Dialog>
         );
       }
       ```

       ```tsx
             else if (result.status === 'skipped_pinned') skipped_pinned++;
             else errors++;
           }
           // POST-APPLY pinned count — must_haves truth #5 (banner accumulates skipped_pinned)
           setResult({ applied, skipped_pinned, errors, rollupStaleAtApply: 0 });
           setApplyState('done');
         }

         return (
           <Dialog open={open} onOpenChange={(o) => !o && onClose()}>
             <DialogContent className="max-w-2xl">
               <DialogHeader>
                 <DialogTitle>
                   Mass edit: "{query}"
                 </DialogTitle>
                 <p className="text-xs text-muted-foreground">
                   {/* PREDICTIVE header — pre-apply pinned count (truth #4) */}
                   {candidates.length} nodes matched
                   {pinnedCount > 0 && <> — {pinnedCount} pinned, will skip</>}
                 </p>
               </DialogHeader>

               {/* EMBEDDING_DISABLED notice — must_haves truth #10. Renders ABOVE the
                   matched-node list. Exact copy is locked: "semantic similarity unavailable
                   — keyword matches only". This closes the loop on 09-01's response.embedding_status
                   field — without this notice, the MASS-01 spec's "with keyword-only fallback"
                   is unverifiable on camera. */}
               {embeddingStatus === 'disabled' && (
                 <div
                   role="status"
                   aria-live="polite"
                   className="mx-1 mb-2 rounded-md border border-amber-200 bg-amber-50 px-3 py-2 text-xs text-amber-900"
                 >
                   semantic similarity unavailable — keyword matches only
                 </div>
               )}

               <ScrollArea className="max-h-[60vh] border rounded-md">
                 {candidates.map(c => (
                   <MatchedNodeRow
                     key={c.uuid}
                     match={c}
                     selected={selectedUuids.has(c.uuid)}
                     onToggle={() => toggle(c.uuid)}
                   />
                 ))}
               </ScrollArea>
               {applyState === 'done' && <MassEditResultBanner />}
               <DialogFooter>
                 {applyState !== 'done' && (
                   <Button variant="outline" onClick={() => setAll(candidates.filter(c => !c.human_pinned).map(c => c.uuid))}>
                     Select all (non-pinned)
                   </Button>
                 )}
                 {applyState !== 'done' && (
                   <Button onClick={approveSelected} disabled={selectedUuids.size === 0 || applyState === 'applying'}>
                     {applyState === 'applying' ? 'Applying…' : `Approve ${selectedUuids.size} selected`}
                   </Button>
                 )}
                 <Button variant="ghost" onClick={onClose}>
                   {applyState === 'done' ? 'Close' : 'Cancel'}
                 </Button>
               </DialogFooter>
             </DialogContent>
           </Dialog>
         );
       }
       ```

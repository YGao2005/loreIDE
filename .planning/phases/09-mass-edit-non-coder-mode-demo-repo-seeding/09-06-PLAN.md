       #!/usr/bin/env bash
       # Validate deletion-incident-2026-02.jsonl is ingestible by Phase 10's filter pipeline.
       # Mirrors the jq filter from constraint-distillation/extraction-prompt.md (or its v1 equivalent).
       set -euo pipefail

       JSONL="$(dirname "${BASH_SOURCE[0]}")/deletion-incident-2026-02.jsonl"

       if [[ ! -f "$JSONL" ]]; then
         echo "ERROR: $JSONL not found" >&2
         exit 1
       fi

       # Step 1: Verify every line is valid JSON
       LINE_COUNT=$(wc -l < "$JSONL")
       if ! jq -ec '.' "$JSONL" > /dev/null; then
         echo "ERROR: $JSONL contains malformed JSON line" >&2
         exit 1
       fi
       echo "[validate] $LINE_COUNT lines, all valid JSON"

       # Step 2: Apply Phase 10's expected filter — extract user/assistant text content
       # NOTE: assistant content is an array of blocks; we extract only the .text-typed blocks
       FILTERED=$(jq -r 'select(.type == "user" or .type == "assistant") |
                         if .type == "user" then .message.content
                         else (.message.content | if type == "array" then map(select(.type == "text") | .text) | join("\n") else . end) end' \
                  "$JSONL" 2>&1)

       if [[ -z "$FILTERED" ]]; then
         echo "ERROR: filter produced empty output — JSONL may not match Phase 10 schema" >&2
         exit 1
       fi
       FILTERED_CHARS=$(echo "$FILTERED" | wc -c)
       echo "[validate] filtered text: $FILTERED_CHARS characters"

         echo "WARNING: priority-shift anchor (compliance-first) not found — Phase 12 narrative may be weakened" >&2
       fi

       echo "[validate] PASS"

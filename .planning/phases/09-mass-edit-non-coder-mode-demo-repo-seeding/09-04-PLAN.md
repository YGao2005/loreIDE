       ```tsx
       'use client';
       import { useState } from 'react';
       import { Button } from './button';

       interface Props {
         children: React.ReactNode;
         onClick: () => void | Promise<void>;
         loading?: boolean;
         confirmation?: 'email-link' | 'modal';
       }

       export function DangerActionButton({ children, onClick, loading, confirmation }: Props) {
         const [showModal, setShowModal] = useState(false);

         async function handleClick() {
           if (confirmation === 'email-link') {
             setShowModal(true);
             // Modal copy: "We'll email you a link to confirm — click within 24 hours."
             // Actual email send fires inside onClick when user confirms in the modal.
             await onClick();
             return;
           }
           if (confirmation === 'modal') {
             if (!window.confirm('Are you sure?')) return;
           }
           await onClick();
         }

         return (
           <Button
             variant="destructive"
             onClick={handleClick}
             disabled={loading}
             className="w-full"
           >
             {loading ? 'Working…' : children}
           </Button>
         );
       }
       ```

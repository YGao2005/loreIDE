    4. **Create `contract-ide/src-tauri/tests/fact_supersession_tests.rs`**:

       ```rust
       //! Adversarial regression harness for fact_engine.
       //! Gated by CI_LLM_LIVE=1 env flag (real claude -p call costs subscription tokens).
       //! Per RESEARCH.md Pattern 5: recall ≥ 80%, precision ≥ 85% on 5 fixtures.
       //!
       //! Run with: CI_LLM_LIVE=1 cargo test --test fact_supersession_tests -- --ignored

       use serde::Deserialize;
       use std::path::PathBuf;

       #[derive(Debug, Deserialize)]
       struct FactFixture {
           scenario: String,
           seed_nodes: Vec<FixtureNode>,
           new_node: FixtureNode,
           expected_invalidated: Vec<String>,
       }

       #[derive(Debug, Deserialize)]
       struct FixtureNode {
           uuid: String,
           node_type: String,
           text: String,
           #[serde(default)]
           scope: Option<String>,
           #[serde(default)]
           applies_when: Option<String>,
           valid_at: String,
       }

       fn load_fixtures() -> Vec<FactFixture> {
           let dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"))
               .join("tests/fixtures/fact_contradictions");
           let mut out = vec![];
           for entry in std::fs::read_dir(&dir).unwrap() {
               let path = entry.unwrap().path();
               if path.extension().and_then(|e| e.to_str()) == Some("json") {
                   let text = std::fs::read_to_string(&path).unwrap();
                   let fx: FactFixture = serde_json::from_str(&text).unwrap();
                   out.push(fx);
               }
           }
           assert!(!out.is_empty(), "No fact_contradictions fixtures found");
           out
       }

       /// Skipped unless CI_LLM_LIVE=1 — exercises real claude -p subprocess.
       /// Ungating: `CI_LLM_LIVE=1 cargo test --test fact_supersession_tests fact_engine -- --ignored`
       #[tokio::test]
       #[ignore]
       async fn fact_engine_recall_at_least_80_percent_precision_at_least_85_percent() {
           if std::env::var("CI_LLM_LIVE").ok().as_deref() != Some("1") {
               eprintln!("[fact_supersession_tests] CI_LLM_LIVE!=1, skipping live LLM call");
               return;
           }

           let fixtures = load_fixtures();
           let mut total_positives = 0usize;
           let mut hits = 0usize;
           let mut false_positives = 0usize;

           // For each fixture: build an in-memory pool with substrate schema, seed
           // the stale node + the new node, run invalidate_contradicted on the new
           // uuid, count hits/false positives against expected_invalidated.
           //
           // NOTE: invalidate_contradicted depends on tauri::AppHandle (for shell
           // and DriftLocks state). The hackathon-pragmatic approach: this test
           // runs the FULL Tauri app in test mode is heavy. Instead, we invoke
           // claude -p directly from the test, build the prompt with the same
           // build_invalidation_prompt, parse with parse_invalidation_response,
           // and check the result. This exercises the LOAD-BEARING components
           // (prompt template + LLM call + verdict parser) without requiring the
           // full app harness. The lock + DB-write paths are unit-tested separately.

           for fx in &fixtures {
               total_positives += fx.expected_invalidated.len();

               // Build the prompt directly via the public API.
               // We need to construct SubstrateNode instances; bring the type into scope.
               let new_node = contract_ide_lib::supersession::types::SubstrateNode {
                   uuid: fx.new_node.uuid.clone(),
                   node_type: fx.new_node.node_type.clone(),
                   text: fx.new_node.text.clone(),
                   scope: fx.new_node.scope.clone(),
                   applies_when: fx.new_node.applies_when.clone(),
                   valid_at: fx.new_node.valid_at.clone(),
                   invalid_at: None,
                   expired_at: None,
                   invalidated_by: None,
               };
               let candidates: Vec<_> = fx.seed_nodes.iter().map(|n| {
                   contract_ide_lib::supersession::types::SubstrateNode {
                       uuid: n.uuid.clone(),
                       node_type: n.node_type.clone(),
                       text: n.text.clone(),
                       scope: n.scope.clone(),
                       applies_when: n.applies_when.clone(),
                       valid_at: n.valid_at.clone(),
                       invalid_at: None,
                       expired_at: None,
                       invalidated_by: None,
                   }
               }).collect();

               let prompt = contract_ide_lib::supersession::prompt::build_invalidation_prompt(
                   &new_node,
                   &candidates,
               );

               // Run claude -p directly (mirrors fact_engine's run_claude_judge).
               let output = tokio::process::Command::new("claude")
                   .arg("-p").arg(&prompt)
                   .arg("--output-format").arg("text")
                   .output().await
                   .expect("claude -p subprocess failed — is claude CLI installed?");
               let raw = String::from_utf8_lossy(&output.stdout).to_string();
               let idxs = contract_ide_lib::supersession::verdict::parse_invalidation_response(&raw)
                   .unwrap_or_default();

               // Map idx → uuid via the candidates vector.
               let invalidated: Vec<String> = idxs.iter()
                   .filter_map(|i| candidates.get(*i).map(|c| c.uuid.clone()))
                   .collect();

               eprintln!("[{}] expected={:?} got={:?}", fx.scenario, fx.expected_invalidated, invalidated);

               for got in &invalidated {
                   if fx.expected_invalidated.contains(got) {
                       hits += 1;
                   } else {
                       false_positives += 1;
                   }
               }
           }

           let recall = if total_positives > 0 {
               hits as f64 / total_positives as f64
           } else { 1.0 };
           let predicted_positives = hits + false_positives;
           let precision = if predicted_positives > 0 {
               hits as f64 / predicted_positives as f64
           } else { 1.0 };

           eprintln!("Recall: {:.2} ({hits}/{total_positives})", recall);
           eprintln!("Precision: {:.2} ({hits}/{predicted_positives})", precision);
           assert!(recall >= 0.80, "Recall {recall:.2} < 0.80 threshold (target: research/intent-supersession baseline)");
           assert!(precision >= 0.85, "Precision {precision:.2} < 0.85 threshold");
       }
       ```

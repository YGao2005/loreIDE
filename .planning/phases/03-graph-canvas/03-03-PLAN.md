**STEP ORDERING IS LOAD-BEARING.** React hooks throw `Invalid hook call` errors during render, and try/catch around hook calls CANNOT catch them (only error boundaries can — and even then, you've already crashed the React tree). The previous draft of this plan tried to defend `useReactFlow()` with try/catch in CommandPalette; that pattern is undefined behavior and must NOT be used. The correct approach is structural: ensure the provider is in scope BEFORE the consumer mounts.

Execute the steps below in the exact order listed. Do all of them in a SINGLE commit so there is no intermediate state in which the palette is mounted but the provider is missing (which would produce the very crash we're avoiding).

**Step 5 — CommandPalette component**. Create `contract-ide/src/components/command-palette/CommandPalette.tsx`.

**KEY: `useReactFlow()` is called UNCONDITIONALLY at the top of the component, NOT wrapped in try/catch.** This works because Step 1 promoted the provider to AppShell — the palette is always mounted as a child of `<ReactFlowProvider>`. React hooks must be called unconditionally per the Rules of Hooks; try/catch around them does not catch React's internal hook-validity errors anyway. If you find yourself wanting to "defend" this hook call, the answer is "ensure the provider is in scope," not "wrap the hook" — see Step 1.

```tsx
import { useCallback, useEffect, useState } from 'react';
import { Command } from 'cmdk';
import { useReactFlow } from '@xyflow/react';
import { useGraphStore } from '@/store/graph';
import { pickAndOpenRepo } from '@/ipc/repo';
import {
  repositoryActions,
  lensActions,
  navigationActions,
} from './actions';
import './commandPalette.css';

interface CommandPaletteProps {
  // Function the palette can call to focus the chat panel — passed by AppShell
  // since chat focus is owned at the layout level (panelRef from AppShell).
  onFocusChat: () => void;
}

export function CommandPalette({ onFocusChat }: CommandPaletteProps) {
  const [open, setOpen] = useState(false);
  const nodes = useGraphStore((s) => s.nodes);
  const setLens = useGraphStore((s) => s.setLens);
  const selectNode = useGraphStore((s) => s.selectNode);

  // Unconditional hook call — provider is in AppShell (Step 1). Do NOT
  // wrap in try/catch: React hook errors are not catchable that way and the
  // attempt produces undefined behavior. If this throws "ReactFlowProvider
  // missing", Step 1 was skipped or reverted — fix the provider scope, not
  // this call site.
  const { getNode, setCenter } = useReactFlow();

  // Cmd+K (or Ctrl+K) toggles the palette. preventDefault BEFORE setOpen so
  // WebKit's "Search in page" cannot fire (RESEARCH §Pitfall 7).
  useEffect(() => {
    const onKey = (e: KeyboardEvent) => {
      if (e.key === 'k' && (e.metaKey || e.ctrlKey)) {
        e.preventDefault();
        setOpen((o) => !o);
      }
      if (e.key === 'Escape') setOpen(false);
    };
    document.addEventListener('keydown', onKey);
    return () => document.removeEventListener('keydown', onKey);
  }, []);

  const close = useCallback(() => setOpen(false), []);

  const handleOpenRepo = useCallback(async () => {
    close();
    try { await pickAndOpenRepo(); } catch (e) { console.warn('open repo failed', e); }
  }, [close]);

  const handleLens = useCallback(async (lens: 'journey' | 'system' | 'ownership') => {
    close();
    await setLens(lens);
  }, [close, setLens]);

  const handleFocusChat = useCallback(() => {
    close();
    onFocusChat();
  }, [close, onFocusChat]);

  const handleJumpToNode = useCallback((uuid: string) => {
    close();
    selectNode(uuid);
    const node = getNode(uuid);
    if (!node) return;
    const cx = node.position.x + (node.measured?.width ?? 160) / 2;
    const cy = node.position.y + (node.measured?.height ?? 60) / 2;
    setCenter(cx, cy, { zoom: 1.5, duration: 600 });
  }, [close, getNode, setCenter, selectNode]);

  return (
    <Command.Dialog open={open} onOpenChange={setOpen} label="Command palette">
      <Command.Input placeholder="Type a command or search nodes…" autoFocus />
      <Command.List>
        <Command.Empty>No results found.</Command.Empty>

        <Command.Group heading="Repository">
          {repositoryActions.map((a) => (
            <Command.Item key={a.id} onSelect={handleOpenRepo}>
              {a.label}
            </Command.Item>
          ))}
        </Command.Group>

        <Command.Group heading="Lens">
          {lensActions.map((a) => (
            <Command.Item key={a.id} onSelect={() => handleLens(a.lens)}>
              {a.label}
            </Command.Item>
          ))}
        </Command.Group>

        <Command.Group heading="Navigation">
          {navigationActions.map((a) => (
            <Command.Item key={a.id} onSelect={handleFocusChat}>
              {a.label}
            </Command.Item>
          ))}
        </Command.Group>

        {nodes.length > 0 && (
          <Command.Group heading="Jump to node">
            {nodes.slice(0, 50).map((n) => (
              <Command.Item
                key={n.uuid}
                value={`${n.name} ${n.level} ${n.kind}`}
                onSelect={() => handleJumpToNode(n.uuid)}
              >
                <span className="text-[10px] uppercase opacity-70 mr-2">{n.level}</span>
                {n.name}
              </Command.Item>
            ))}
          </Command.Group>
        )}
      </Command.List>
    </Command.Dialog>
  );
}
```

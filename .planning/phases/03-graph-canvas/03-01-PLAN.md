```ts
import { create } from 'zustand';
import type { ContractNode } from '@/ipc/types';
import { getNodes } from '@/ipc/nodes';

export type LensId = 'journey' | 'system' | 'ownership';

interface GraphState {
  nodes: ContractNode[];
  selectedNodeUuid: string | null;
  currentLens: LensId;
  selectNode: (uuid: string | null) => void;
  setLens: (lens: LensId) => void;
  refreshNodes: () => Promise<void>;
}

export const useGraphStore = create<GraphState>((set) => ({
  nodes: [],
  selectedNodeUuid: null,
  currentLens: 'journey',
  selectNode: (selectedNodeUuid) => set({ selectedNodeUuid }),
  setLens: (currentLens) => set({ currentLens }),
  refreshNodes: async () => {
    const rows = await getNodes();
    set({ nodes: rows });
  },
}));
```

        UI:   'border-blue-500 bg-blue-50 text-blue-900',
        API:  'border-violet-500 bg-violet-50 text-violet-900',
        data: 'border-amber-500 bg-amber-50 text-amber-900',
        job:  'border-emerald-500 bg-emerald-50 text-emerald-900',
        unknown: 'border-slate-400 bg-slate-50 text-slate-900',
      },
      state: {
        healthy:  '',
        drifted:  'ring-2 ring-red-500 animate-pulse',
        untested: 'opacity-70',
      },
      canonical: {
        true:  '',
        false: 'border-dashed opacity-60',
      },
    },
    defaultVariants: { kind: 'unknown', state: 'healthy', canonical: 'true' },
  }
);

// Map a ContractNode.kind string (free-form from sidecar) to a known variant.
// Anything unrecognised gets the neutral 'unknown' slot rather than crashing
// the cva variant lookup.
export function normalizeKind(kind: string): 'UI' | 'API' | 'data' | 'job' | 'unknown' {
  switch (kind) {
    case 'UI':
    case 'API':
    case 'data':
    case 'job':
      return kind;
    default:
      return 'unknown';
  }
}

// Phase 3 has no real drift state yet (Phase 7 owns DRIFT-01). Default every
// node to 'healthy' for now; Phase 7 will populate this from drift_state.
export type NodeHealthState = 'healthy' | 'drifted' | 'untested';

**Create `contract-ide/src/components/graph/GraphCanvasInner.tsx`** — the actual ReactFlow mount. Reads `nodes` from `useGraphStore` and maps each `ContractNode` (the data type from `@/ipc/types`) to a React Flow `Node`. Phase 3 Plan 1 has NO edges yet (Plan 03-02 adds `get_edges`); pass empty `edges`. Hand-position nodes in a simple grid for now — Plan 03-02 introduces the parentId hierarchy.

**CRITICAL — fitView + onlyRenderVisibleElements race (RESEARCH §Pitfall 9):** With virtualization on day-one, no nodes are measured at first mount → a default `fitView` prop fits to an empty bounding box → blank viewport on repo open. Fix:

1. Set an explicit `defaultViewport={{ x: 0, y: 0, zoom: 0.8 }}` so the initial frame is non-blank even before any node measures.
2. DROP the bare `fitView` prop on `<ReactFlow>` (it would race with virtualization).
3. After mount, run `useReactFlow().fitView({ padding: 0.2, minZoom: 0.5, duration: 400 })` from a `useEffect` keyed on the row count, deferred via `setTimeout(..., 100)` so React Flow has at least one paint to measure visible nodes.

This is simpler than wiring `onNodesChange` for `change.type === 'dimensions'` and produces an indistinguishable result for the demo. Document the choice in a code comment so future plans don't "helpfully" add the bare `fitView` back.

```tsx
import { useEffect, useMemo } from 'react';
import {
  ReactFlow,
  Background,
  Controls,
  useReactFlow,
  type Node,
  type Edge,
} from '@xyflow/react';
import { useGraphStore } from '@/store/graph';
import { nodeTypes } from './nodeTypes';
import type { ContractNodeData } from './ContractNode';
import type { ContractNode as RowContractNode } from '@/ipc/types';

// Phase 3 Plan 1: simple grid layout. Plan 03-02 replaces this with a
// proper parentId + extent:'parent' hierarchical layout (RESEARCH §Pattern 2).
// Hand-curated demo seeds (Phase 9) will fit easily inside any layout we pick.
function rowToFlowNode(row: RowContractNode, index: number): Node<ContractNodeData> {
  const col = index % 6;
  const rowIdx = Math.floor(index / 6);
  return {
    id: row.uuid,
    type: 'contract',
    position: { x: col * 180, y: rowIdx * 100 },
    data: {
      name: row.name,
      kind: row.kind,
      state: 'healthy', // Phase 7 (DRIFT-01) populates real drift state
      isCanonical: row.is_canonical,
      level: row.level,
    },
  };
}

export function GraphCanvasInner() {
  // SELECTOR — only re-render when the rows ref changes (refreshNodes()
  // replaces it wholesale). Do NOT subscribe to s.selectedNodeUuid here;
  // the canvas should not re-render on selection.
  const rows = useGraphStore((s) => s.nodes);
  const { fitView } = useReactFlow();

  const nodes = useMemo<Node<ContractNodeData>[]>(
    () => rows.map(rowToFlowNode),
    [rows]
  );
  // Empty edges in Plan 03-01; Plan 03-02 wires get_edges.
  const edges = useMemo<Edge[]>(() => [], []);

  // RESEARCH §Pitfall 9 — `fitView` prop races with `onlyRenderVisibleElements`
  // because virtualized nodes have no measured dimensions on first paint, so
  // React Flow fits to an empty bbox = blank viewport. Defer fitView to after
  // first paint via setTimeout so at least the initial visible-window nodes
  // have measured. Re-run when row count changes (lens switch / scan).
  useEffect(() => {
    if (rows.length === 0) return;
    const t = setTimeout(() => {
      fitView({ padding: 0.2, minZoom: 0.5, duration: 400 });
    }, 100);
    return () => clearTimeout(t);
  }, [rows.length, fitView]);

  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      nodeTypes={nodeTypes}              // module-level const (Pitfall 1)
      onlyRenderVisibleElements          // GRAPH-03 — DAY ONE per STATE.md
      minZoom={0.1}
      maxZoom={2}
      // NOTE: bare `fitView` prop intentionally OMITTED — see Pitfall 9 above.
      // The deferred fitView in useEffect handles initial framing without
      // racing with virtualization. defaultViewport supplies the first frame.
      defaultViewport={{ x: 0, y: 0, zoom: 0.8 }}
      proOptions={{ hideAttribution: true }}
    >
      <Background />
      <Controls />
    </ReactFlow>
  );
}
```

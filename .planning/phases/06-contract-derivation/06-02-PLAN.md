**A. `contract-ide/src/ipc/derive.ts`** — new file mirroring the shape of `src/ipc/mcp.ts` from Plan 05-01:

```typescript
import { listen, type UnlistenFn } from '@tauri-apps/api/event';
import { invoke } from '@tauri-apps/api/core';

export type DeriveStatus =
  | 'running'
  | 'done'
  | 'skipped'
  | 'skipped-pinned'
  | 'error';

/**
 * Event payload emitted by the Plan 06-01 Rust command
 * (serde rename_all = "camelCase"). Fields must match
 * src-tauri/src/commands/derive.rs `DeriveProgress` exactly.
 */
export interface DeriveProgress {
  uuid: string;
  status: DeriveStatus;
  message?: string;
}

/** Fire-and-forget invoker. Resolves immediately; real progress arrives via
 *  the `derive:progress` event stream subscribed in AppShell. */
export async function deriveContracts(uuids: string[]): Promise<void> {
  await invoke<void>('derive_contracts', { uuids });
}

/** Subscribe to per-node derivation progress. Returns an UnlistenFn — call it
 *  from a useEffect cleanup to detach. Exactly one subscriber should exist
 *  at any time (mounted in AppShell); component-level listeners would
 *  multiply event traffic on a 500-node graph. */
export async function subscribeDeriveProgress(
  onEvent: (payload: DeriveProgress) => void,
): Promise<UnlistenFn> {
  return listen<DeriveProgress>('derive:progress', (tauriEvent) => {
    onEvent(tauriEvent.payload);
  });
}
```

**B. `contract-ide/src/store/derive.ts`** — Zustand store keyed by uuid. Verify Zustand is already a dep (it is — Phase 1 `01-03-SUMMARY.md` / `store/graph.ts` use it). If not, install with `npm install zustand` in `contract-ide/`.

```typescript
import { create } from 'zustand';
import type { DeriveProgress } from '@/ipc/derive';

interface DeriveState {
  /** Last-seen progress keyed by node uuid. */
  progress: Record<string, DeriveProgress>;
  setProgress: (event: DeriveProgress) => void;
  clearProgress: (uuid: string) => void;
}

export const useDeriveStore = create<DeriveState>((set) => ({
  progress: {},
  setProgress: (event) =>
    set((s) => ({ progress: { ...s.progress, [event.uuid]: event } })),
  clearProgress: (uuid) =>
    set((s) => {
      const next = { ...s.progress };
      delete next[uuid];
      return { progress: next };
    }),
}));
```

```tsx
import { useEffect } from 'react';
import { Button } from '@/components/ui/button';
import { useGraphStore } from '@/store/graph';
import { useDeriveStore } from '@/store/derive';
import { deriveContracts } from '@/ipc/derive';

export function Inspector() {
  // Adapt the selector names to whatever store/graph.ts actually exports.
  const selectedUuid = useGraphStore((s) => s.selectedNodeUuid);
  const node = useGraphStore((s) =>
    selectedUuid ? s.nodes.find((n) => n.uuid === selectedUuid) : undefined,
  );
  const progress = useDeriveStore((s) =>
    selectedUuid ? s.progress[selectedUuid] : undefined,
  );
  const clearProgress = useDeriveStore((s) => s.clearProgress);

  // On done -> refresh graph so contract_body reflects the new LLM output.
  useEffect(() => {
    if (!progress || !selectedUuid) return;
    if (progress.status === 'done') {
      useGraphStore.getState().refreshNodes();
    }
    // Auto-clear non-error terminal states after 3s so the label doesn't
    // stay stale across multiple clicks.
    if (['done', 'skipped', 'skipped-pinned'].includes(progress.status)) {
      const t = setTimeout(() => clearProgress(selectedUuid), 3000);
      return () => clearTimeout(t);
    }
  }, [progress, selectedUuid, clearProgress]);

  if (!node) {
    return <div className="text-muted-foreground">Select a node to inspect.</div>;
  }

  const isRunning = progress?.status === 'running';
  const label =
    !progress
      ? null
      : progress.status === 'running'
        ? 'Deriving…'
        : progress.status === 'done'
          ? 'Derived'
          : progress.status === 'skipped'
            ? 'Skipped (no changes)'
            : progress.status === 'skipped-pinned'
              ? 'Skipped (pinned)'
              : `Error: ${progress.message ?? 'unknown'}`;

  return (
    <div className="flex flex-col gap-3 p-3">
      {/* Keep whatever existing Phase 1 stub content already lives here
          (e.g. a heading rendering node.name, empty states, etc.). */}
      <div className="flex items-center gap-2">
        <strong>{node.name}</strong>
        {node.human_pinned ? (
          <span
            className="rounded bg-amber-100 px-1.5 py-0.5 text-xs text-amber-900"
            title="Human-pinned — derivation will skip this node"
          >
            pinned
          </span>
        ) : null}
      </div>
      <div className="flex items-center gap-2">
        <Button
          size="sm"
          disabled={isRunning}
          onClick={() => deriveContracts([node.uuid])}
        >
          {isRunning ? 'Deriving…' : 'Derive'}
        </Button>
        {label && progress?.status !== 'running' ? (
          <span className="text-xs text-muted-foreground">{label}</span>
        ) : null}
      </div>
      {/* Contract body preview — safe fallback if the existing stub didn't
          render it. Small enough that replacing only this section is OK. */}
      {node.contract_body ? (
        <pre className="whitespace-pre-wrap text-xs text-muted-foreground">
          {node.contract_body}
        </pre>
      ) : (
        <div className="text-xs italic text-muted-foreground">
          No contract yet — click Derive to generate one.
        </div>
      )}
    </div>
  );
}
```

```rust
use dashmap::DashMap;
use std::sync::Arc;
use tokio::sync::Mutex;

/// Per-node serialization guard. One mutex per node UUID, lazily inserted on
/// first drift event. Never garbage-collected (demo scale < 1k nodes; the
/// memory cost is 48 bytes per entry).
///
/// CRITICAL: Uses `tokio::sync::Mutex` (NOT `std::sync::Mutex`). The guard
/// is held across `.await` points (DB queries). std::sync::Mutex is not
/// Send-across-await and panics the scheduler under load.
///
/// Phase 8's PostToolUse hook will reuse this same map via
/// `DriftLocks::for_uuid(uuid)` → guaranteed no race between watcher and hook.
#[derive(Default)]
pub struct DriftLocks(pub DashMap<String, Arc<Mutex<()>>>);

impl DriftLocks {
    pub fn for_uuid(&self, uuid: &str) -> Arc<Mutex<()>> {
        self.0
            .entry(uuid.to_string())
            .or_insert_with(|| Arc::new(Mutex::new(())))
            .clone()
    }
}
```

Create `contract-ide/src-tauri/src/drift/watcher.rs` wrapping `notify::RecommendedWatcher` with a `refresh(app, repo_path, nodes)` API that (re)registers per-file watches based on the current set of node → code_ranges pairs. Follow RESEARCH.md §Pattern 1 closely.

**Structure:**

```rust
use notify::{RecommendedWatcher, RecursiveMode, Watcher, Event, EventKind};
use std::path::{Path, PathBuf};
use std::sync::Mutex;
use std::collections::HashMap;
use tauri::AppHandle;

pub struct SourceWatcher {
    inner: Mutex<Option<RecommendedWatcher>>,
    /// canonicalized absolute file path → set of node uuids referencing it
    path_to_uuids: Mutex<HashMap<PathBuf, Vec<String>>>,
}

impl SourceWatcher {
    pub fn new() -> Self {
        Self {
            inner: Mutex::new(None),
            path_to_uuids: Mutex::new(HashMap::new()),
        }
    }

    /// Rebuild the watcher registration from `nodes = Vec<(uuid, Vec<rel_file_path>)>`.
    /// Called from `open_repo` and every `refresh_nodes` (Plan 07-02 wires these).
    pub fn refresh(
        &self,
        app: AppHandle,
        repo_path: &Path,
        nodes: &[(String, Vec<String>)],
    ) -> anyhow::Result<()> {
        // Build canonical absolute-path → uuids map.
        let mut map: HashMap<PathBuf, Vec<String>> = HashMap::new();
        for (uuid, files) in nodes {
            for rel in files {
                // Pitfall 9: filter out any path pointing INTO .contracts/ — a
                // node's code_ranges accidentally pointing at its own sidecar
                // would cause double-trigger with the existing sidecar watcher.
                if rel.contains(".contracts/") || rel.starts_with(".contracts") {
                    continue;
                }
                let abs = repo_path.join(rel);
                // Pitfall 3: canonicalize so FSEvents-reported paths match.
                // If the file doesn't exist yet, skip (can't canonicalize a
                // non-existent path on macOS).
                let canon = match std::fs::canonicalize(&abs) {
                    Ok(p) => p,
                    Err(_) => continue,
                };
                map.entry(canon).or_default().push(uuid.clone());
            }
        }
        for v in map.values_mut() {
            v.sort();
            v.dedup();
        }

        // Snapshot for the closure.
        let snapshot = map.clone();
        let app_cb = app.clone();

        let mut watcher = notify::recommended_watcher(move |res: notify::Result<Event>| {
            let Ok(event) = res else { return };
            // Pitfall 4: accept Modify / Create / Remove — editors save
            // atomically (temp + rename) which produces Create+Remove, not
            // Modify. Filter out metadata/access-only events.
            match event.kind {
                EventKind::Modify(_) | EventKind::Create(_) | EventKind::Remove(_) => {}
                _ => return,
            }
            for path in &event.paths {
                // Also canonicalize the event path (Pitfall 3 defence-in-depth).
                let key = std::fs::canonicalize(path).unwrap_or_else(|_| path.clone());
                if let Some(uuids) = snapshot.get(&key) {
                    for uuid in uuids {
                        let app2 = app_cb.clone();
                        let uuid2 = uuid.clone();
                        tauri::async_runtime::spawn(async move {
                            crate::drift::engine::compute_and_emit(app2, &uuid2).await;
                        });
                    }
                }
            }
        })?;

        // NonRecursive: watching individual files, not directories. Parent-dir
        // recursive watch is an Open Question 2 escalation for > 1000 files;
        // at hackathon scale (< 200) individual-file watches are cheap.
        for path in map.keys() {
            // Swallow per-path errors (some files may vanish between canonicalize
            // and watch) — an Err on one file must not abort the whole refresh.
            let _ = watcher.watch(path, RecursiveMode::NonRecursive);
        }

        *self.inner.lock().unwrap() = Some(watcher);
        *self.path_to_uuids.lock().unwrap() = map;
        Ok(())
    }
}

impl Default for SourceWatcher {
    fn default() -> Self {
        Self::new()
    }
}
```

**Key correctness notes** (enforce by comment in the code):

- `inner: Mutex<Option<RecommendedWatcher>>` — `std::sync::Mutex` is correct HERE because the guard is never held across `.await` (refresh is a sync fn). The notify watcher itself drops cleanly when replaced.
- **Pitfall 2** (FSEvents coalescing): documented inline — don't rely on event count, rely on event happened. The SC 2 test in 07-04 asserts final `drift_state` rows, not event counts.
- **Pitfall 7** (watcher not rebuilt on new code_ranges): `refresh()` is called from BOTH `open_repo` and `refresh_nodes` in Plan 07-02 — a fresh node starts being watched within the 2s sidecar-refresh window.
- `notify::recommended_watcher` on macOS picks `FSEvents` automatically (our `macos_fsevent` feature is enabled). No need for explicit backend config.

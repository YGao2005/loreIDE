**A. Rust `refresh_nodes` command (in `commands/repo.rs`).**

```rust
#[tauri::command]
pub async fn refresh_nodes(
    app: tauri::AppHandle,
    paths: Vec<String>,
) -> Result<crate::db::scanner::ScanResult, String> {
    // Only re-parse files under .contracts/ that still exist. Filter out non-.md
    // and non-existent paths (delete events fire with the old path too).
    let mut updated: u32 = 0;
    let mut errors: Vec<String> = Vec::new();

    let instances = app.state::<tauri_plugin_sql::DbInstances>();
    let db_map = instances.0.read().await;
    let db = db_map.get("sqlite:contract-ide.db").ok_or("DB not loaded")?;

    for p in paths {
        let path = std::path::Path::new(&p);
        if path.extension().and_then(|x| x.to_str()) != Some("md") {
            continue;
        }
        if !path.exists() {
            // TODO(Phase 7): propagate a delete to SQLite via DELETE FROM nodes WHERE uuid = ?.
            // For Phase 2 the watcher focuses on create/modify; deletes are a reconciliation
            // concern that belongs with drift detection.
            continue;
        }
        match std::fs::read_to_string(path) {
            Ok(content) => match crate::sidecar::frontmatter::parse_sidecar(&content) {
                Ok((fm, body)) => {
                    if let Err(e) = crate::db::scanner::upsert_node_pub(db, &fm, &body).await {
                        errors.push(format!("Upsert {} failed: {e}", fm.uuid));
                    } else {
                        updated += 1;
                    }
                }
                Err(e) => errors.push(format!("Parse {p}: {e}")),
            },
            Err(e) => errors.push(format!("Read {p}: {e}")),
        }
    }

    Ok(crate::db::scanner::ScanResult {
        node_count: updated,
        error_count: errors.len() as u32,
        errors,
    })
}
```

**C. JS watcher wrapper — `src/ipc/watcher.ts`.**

```typescript
import { watch, type UnwatchFn, type WatchEvent } from '@tauri-apps/plugin-fs';
import { invoke } from '@tauri-apps/api/core';

let unwatchFn: UnwatchFn | null = null;

export interface WatcherHandlers {
  /** Called after Rust has re-upserted. Consumers should refetch graph state here. */
  onRefreshed: (affectedCount: number, errors: string[]) => void;
}

/**
 * Start the .contracts/ watcher. Debounced at 2000ms by tauri-plugin-fs
 * (matches DATA-03's "within 2 seconds" target). Recursive so nested
 * .contracts/ directories Phase 9 may add still fire events.
 *
 * Must be called AFTER open_repo's scan_contracts_dir completes
 * (Pitfall 4 — scan/watch race otherwise possible).
 */
export async function startContractsWatcher(
  contractsPath: string,
  handlers: WatcherHandlers,
): Promise<void> {
  await stopContractsWatcher();
  unwatchFn = await watch(
    contractsPath,
    async (event: WatchEvent) => {
      // Filter to .md paths only; tauri-plugin-fs already debounces.
      const mdPaths = event.paths.filter((p) => p.endsWith('.md'));
      if (mdPaths.length === 0) return;
      try {
        const result = await invoke<{
          nodeCount: number;
          errorCount: number;
          errors: string[];
        }>('refresh_nodes', { paths: mdPaths });
        handlers.onRefreshed(result.nodeCount, result.errors);
      } catch (e) {
        handlers.onRefreshed(0, [String(e)]);
      }
    },
    { recursive: true, delayMs: 2000 },
  );
}

export async function stopContractsWatcher(): Promise<void> {
  if (unwatchFn) {
    await unwatchFn();
    unwatchFn = null;
  }
}
```

In `src/ipc/repo.ts`, extend `pickAndOpenRepo()` to accept a refresh callback and start the watcher after the scan succeeds:

```typescript
export async function pickAndOpenRepo(
  onRefreshed?: (count: number, errors: string[]) => void,
): Promise<ScanResult | null> {
  const folder = await open({ multiple: false, directory: true });
  if (!folder || typeof folder !== 'string') return null;
  const result = await invoke<ScanResult>('open_repo', { repoPath: folder });
  if (onRefreshed) {
    const contractsPath = `${folder}/.contracts`;
    // Only start the watcher if .contracts exists (avoids a plugin error on empty repos).
    await startContractsWatcher(contractsPath, {
      onRefreshed: (count, errors) => onRefreshed(count, errors),
    }).catch((e) => {
      // Watcher failure is non-fatal — scan already succeeded.
      console.warn('watcher start failed:', e);
    });
  }
  return result;
}
```

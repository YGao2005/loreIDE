Walk `<repo_path>/.contracts/` recursively via `walkdir`, parse every `.md` via `crate::sidecar::frontmatter::parse_sidecar`, track `seen_uuids: HashSet<String>` and skip-with-error on duplicates (DATA-02 success criterion 5), upsert via `ON CONFLICT(uuid) DO UPDATE` so rename/move of the .md file itself or the source it points to never mints a new UUID (DATA-04). DATA-05 ghost-ref generation is explicitly deferred to Phase 3 — leave a `// TODO(Phase 3): rebuild_ghost_refs()` comment, do NOT implement here.

Key concerns baked in:

- Wrap the synchronous walkdir+read_to_string in `tauri::async_runtime::spawn_blocking` (Pitfall §Anti-Patterns 5 in 02-RESEARCH.md) so we don't block the Tokio executor on filesystem I/O.
- **Collect parse/read failures inside the spawn_blocking closure**. Do NOT drop them. Return `(parsed: Vec<(fm, body, path)>, parse_errors: Vec<String>)` from the closure; merge `parse_errors` into the outer `errors` vec before returning `ScanResult`.
- Return a `ScanResult { node_count: u32, error_count: u32, errors: Vec<String> }` so the frontend can surface duplicates, parse failures, and read failures.
- Access the DB pool via `app.state::<tauri_plugin_sql::DbInstances>()` and hold the read guard for the duration of the queries (Pitfall 1).
- Use the DB URL `"sqlite:contract-ide.db"` (identical to Phase 1's `tauri.conf.json` preload key).
- For `is_canonical`: scanner seeds every scanned sidecar as `is_canonical=1`. Ghost-ref generation (is_canonical=0 rows derived from node_flows) is deferred to Phase 3.
- The upsert should set `name = fm.code_ranges[0].file basename OR fm.uuid` — the name column is NOT NULL in v1 schema (see 01-02-SUMMARY.md SQL).
- Map `fm.code_ranges` → `serde_json::to_string(&fm.code_ranges)` stored in the `code_ranges` TEXT column.
- **Name the public upsert `upsert_node_pub`**. This is the single canonical name the rest of the crate imports (Plan 02-02 Task 2's `write_contract` and Plan 02-03's `refresh_nodes` both hardcode `crate::db::scanner::upsert_node_pub`). Do NOT introduce a separate private `upsert_node` + re-export — keep one `pub fn upsert_node_pub` to avoid a naming-ambiguity trap.

```rust
use std::collections::HashSet;
use std::path::Path;
use walkdir::WalkDir;
use serde::Serialize;
use sqlx::{Pool, Sqlite};
use tauri_plugin_sql::DbInstances;
use crate::sidecar::frontmatter::{parse_sidecar, ContractFrontmatter};

#[derive(Debug, Serialize, Clone)]
#[serde(rename_all = "camelCase")]
pub struct ScanResult {
    pub node_count: u32,
    pub error_count: u32,
    pub errors: Vec<String>,
}

pub async fn scan_contracts_dir(
    app: &tauri::AppHandle,
    repo_path: &Path,
) -> anyhow::Result<ScanResult> {
    let contracts_dir = repo_path.join(".contracts");
    if !contracts_dir.exists() {
        return Ok(ScanResult { node_count: 0, error_count: 0, errors: vec![] });
    }

    // Phase 1: walk + parse synchronously under spawn_blocking. Collect parse/read
    // errors into a dedicated vec (DO NOT drop them) and merge them into the outer
    // errors vec after the blocking step returns.
    let contracts_dir_owned = contracts_dir.clone();
    let (parsed, parse_errors): (
        Vec<(ContractFrontmatter, String, String)>,
        Vec<String>,
    ) = tauri::async_runtime::spawn_blocking(move || {
        let mut out = Vec::new();
        let mut errs = Vec::<String>::new();
        for entry in WalkDir::new(&contracts_dir_owned)
            .follow_links(false)
            .into_iter()
            .filter_map(|e| e.ok())
            .filter(|e| e.path().extension().and_then(|x| x.to_str()) == Some("md"))
        {
            let path_display = entry.path().display().to_string();
            match std::fs::read_to_string(entry.path()) {
                Ok(content) => match parse_sidecar(&content) {
                    Ok((fm, body)) => out.push((fm, body, path_display)),
                    Err(e) => errs.push(format!("Parse {path_display}: {e}")),
                },
                Err(e) => errs.push(format!("Read {path_display}: {e}")),
            }
        }
        (out, errs)
    })
    .await??;

    // Phase 2: duplicate detection + upsert sequentially.
    //
    // NOTE: Phase 2 keeps scan best-effort per-sidecar rather than wrapping in a
    // single transaction. A partial scan (one bad row skipped, rest upserted)
    // leaves SQLite in a usable state — which is what the watcher (02-03) also
    // produces. A future phase can add BEGIN IMMEDIATE / COMMIT around the loop
    // if atomicity becomes observable (e.g. the graph renders half-scanned
    // state visibly). For Phase 2 the UI surfaces ScanResult.errors instead.
    let instances = app.state::<DbInstances>();
    let db_map = instances.0.read().await;
    let db = db_map
        .get("sqlite:contract-ide.db")
        .ok_or_else(|| anyhow::anyhow!("sqlite:contract-ide.db pool not loaded"))?;

    let mut seen = HashSet::<String>::new();
    let mut errors = parse_errors; // merge walk-phase errors first
    let mut n: u32 = 0;

    for (fm, body, path) in parsed {
        if !seen.insert(fm.uuid.clone()) {
            errors.push(format!("Duplicate UUID {} in {}", fm.uuid, path));
            continue;
        }
        if let Err(e) = upsert_node_pub(db, &fm, &body).await {
            errors.push(format!("Upsert {} failed: {e}", fm.uuid));
            continue;
        }
        n += 1;
    }

    Ok(ScanResult {
        node_count: n,
        error_count: errors.len() as u32,
        errors,
    })
}

/// The canonical name every other module uses. Do not introduce a private
/// `upsert_node` variant — the whole crate points at this one symbol.
pub async fn upsert_node_pub(
    db: &Pool<Sqlite>,
    fm: &ContractFrontmatter,
    body: &str,
) -> anyhow::Result<()> {
    let code_ranges_json = serde_json::to_string(&fm.code_ranges)?;
    let name = fm
        .code_ranges
        .first()
        .map(|r| {
            std::path::Path::new(&r.file)
                .file_name()
                .and_then(|s| s.to_str())
                .unwrap_or(&fm.uuid)
                .to_string()
        })
        .unwrap_or_else(|| fm.uuid.clone());
    let human_pinned_i = if fm.human_pinned { 1i64 } else { 0i64 };

    // 1) nodes upsert.
    sqlx::query(
        r#"
        INSERT INTO nodes (
            uuid, level, name, kind, code_ranges, parent_uuid,
            is_canonical, code_hash, contract_hash, human_pinned,
            route, derived_at, contract_body, updated_at
        )
        VALUES (?1, ?2, ?3, ?4, ?5, ?6, 1, ?7, ?8, ?9, ?10, ?11, ?12, datetime('now'))
        ON CONFLICT(uuid) DO UPDATE SET
            level = excluded.level,
            kind = excluded.kind,
            code_ranges = excluded.code_ranges,
            parent_uuid = excluded.parent_uuid,
            code_hash = excluded.code_hash,
            contract_hash = excluded.contract_hash,
            human_pinned = excluded.human_pinned,
            route = excluded.route,
            derived_at = excluded.derived_at,
            contract_body = excluded.contract_body,
            updated_at = datetime('now')
        "#,
    )
    .bind(&fm.uuid)
    .bind(&fm.level)
    .bind(&name)
    .bind(&fm.kind)
    .bind(&code_ranges_json)
    .bind(&fm.parent)
    .bind(&fm.code_hash)
    .bind(&fm.contract_hash)
    .bind(human_pinned_i)
    .bind(&fm.route)
    .bind(&fm.derived_at)
    .bind(body)
    .execute(db)
    .await?;

    // 2) edges: one row per neighbor UUID (Phase 2 success criterion 1 —
    //    "populate nodes, edges, node_flows"). We clear existing outgoing
    //    edges from this node first so a removed neighbor disappears on rescan,
    //    then insert the current set.
    sqlx::query("DELETE FROM edges WHERE from_uuid = ?1")
        .bind(&fm.uuid)
        .execute(db)
        .await?;
    for neighbor in &fm.neighbors {
        sqlx::query(
            r#"
            INSERT OR IGNORE INTO edges (from_uuid, to_uuid)
            VALUES (?1, ?2)
            "#,
        )
        .bind(&fm.uuid)
        .bind(neighbor)
        .execute(db)
        .await?;
    }

    // 3) node_flows: a node's flow membership comes from (a) its parent
    //    (primary flow) and (b) its route when present (secondary flow keyed
    //    by route string). DATA-05 ghost-ref generation is Phase 3 — here we
    //    only materialise canonical membership. Wipe + reinsert on rescan so
    //    a dropped parent/route is reflected.
    sqlx::query("DELETE FROM node_flows WHERE node_uuid = ?1")
        .bind(&fm.uuid)
        .execute(db)
        .await?;
    if let Some(parent) = &fm.parent {
        sqlx::query(
            r#"
            INSERT OR IGNORE INTO node_flows (node_uuid, flow_uuid)
            VALUES (?1, ?2)
            "#,
        )
        .bind(&fm.uuid)
        .bind(parent)
        .execute(db)
        .await?;
    }
    if let Some(route) = &fm.route {
        sqlx::query(
            r#"
            INSERT OR IGNORE INTO node_flows (node_uuid, flow_uuid)
            VALUES (?1, ?2)
            "#,
        )
        .bind(&fm.uuid)
        .bind(route)
        .execute(db)
        .await?;
    }

    // TODO(Phase 3): rebuild_ghost_refs() — derive is_canonical=0 rows keyed
    // by canonical_uuid from node_flows membership. Deferred per RESEARCH.md
    // Open Question 4 and ROADMAP Phase 3 requirements.

    Ok(())
}
```

**A. `src-tauri/src/commands/contracts.rs` — write_contract (single writer).**

```rust
#[tauri::command]
pub async fn write_contract(
    app: tauri::AppHandle,
    repo_path: String,
    uuid: String,
    frontmatter: crate::sidecar::frontmatter::ContractFrontmatter,
    body: String,
) -> Result<(), String> {
    // Serialize the .md string.
    let contents = crate::sidecar::frontmatter::write_sidecar(&frontmatter, &body)
        .map_err(|e| e.to_string())?;

    // Atomic write: temp file + rename. Both in .contracts/ so rename is same-fs.
    let contracts_dir = std::path::Path::new(&repo_path).join(".contracts");
    std::fs::create_dir_all(&contracts_dir).map_err(|e| e.to_string())?;
    let target = contracts_dir.join(format!("{uuid}.md"));
    let tmp = contracts_dir.join(format!(".{uuid}.md.tmp"));

    std::fs::write(&tmp, contents).map_err(|e| e.to_string())?;
    std::fs::rename(&tmp, &target).map_err(|e| e.to_string())?;

    // Re-upsert into SQLite so the cache is immediately in sync (watcher in 02-03
    // will also fire, but we don't want a race between 'write' and 'graph refresh').
    //
    // Call the ONE canonical public symbol — upsert_node_pub — established in
    // Task 1. No ambiguous naming here.
    let instances = app.state::<tauri_plugin_sql::DbInstances>();
    let db_map = instances.0.read().await;
    let db = db_map.get("sqlite:contract-ide.db").ok_or("DB not loaded")?;
    crate::db::scanner::upsert_node_pub(db, &frontmatter, &body)
        .await
        .map_err(|e| e.to_string())?;

    Ok(())
}
```

```rust
    pub derived_at: Option<String>,
    pub contract_body: Option<String>,
    pub tags: Vec<String>,
}

#[tauri::command]
pub async fn get_nodes(
    app: tauri::AppHandle,
    level: Option<String>,
    parent_uuid: Option<String>,
) -> Result<Vec<ContractNode>, String> {
    let instances = app.state::<tauri_plugin_sql::DbInstances>();
    let db_map = instances.0.read().await;
    let db = db_map.get("sqlite:contract-ide.db").ok_or("DB not loaded")?;

    let rows: Vec<sqlx::sqlite::SqliteRow> = sqlx::query(
        r#"
        SELECT uuid, level, name, kind, code_ranges, parent_uuid, is_canonical,
               code_hash, contract_hash, human_pinned, route, derived_at,
               contract_body, tags
        FROM nodes
        WHERE (?1 IS NULL OR level = ?1)
          AND (?2 IS NULL OR parent_uuid = ?2)
        "#,
    )
    .bind(level)
    .bind(parent_uuid)
    .fetch_all(db)
    .await
    .map_err(|e| e.to_string())?;

    use sqlx::Row;
    let mut out = Vec::with_capacity(rows.len());
    for r in rows {
        let code_ranges_str: Option<String> = r.try_get("code_ranges").unwrap_or(None);
        let code_ranges: Vec<CodeRange> = code_ranges_str
            .as_deref()
            .map(|s| serde_json::from_str(s).unwrap_or_default())
            .unwrap_or_default();
        let tags_str: Option<String> = r.try_get("tags").unwrap_or(None);
        let tags: Vec<String> = tags_str
            .as_deref()
            .map(|s| serde_json::from_str(s).unwrap_or_default())
            .unwrap_or_default();
        out.push(ContractNode {
            uuid: r.try_get("uuid").map_err(|e| e.to_string())?,
            level: r.try_get("level").map_err(|e| e.to_string())?,
            name: r.try_get("name").map_err(|e| e.to_string())?,
            kind: r.try_get("kind").map_err(|e| e.to_string())?,
            code_ranges,
            parent_uuid: r.try_get("parent_uuid").ok(),
            is_canonical: r.try_get::<i64, _>("is_canonical").map(|v| v != 0).unwrap_or(true),
            code_hash: r.try_get("code_hash").ok(),
            contract_hash: r.try_get("contract_hash").ok(),
            human_pinned: r.try_get::<i64, _>("human_pinned").map(|v| v != 0).unwrap_or(false),
            route: r.try_get("route").ok(),
            derived_at: r.try_get("derived_at").ok(),
            contract_body: r.try_get("contract_body").ok(),
            tags,
        });
    }
    Ok(out)
}
```

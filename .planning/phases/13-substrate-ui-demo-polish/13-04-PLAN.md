import { useGraphStore } from '@/store/graph';
import { useSubstrateStore } from '@/store/substrate';
import { useDriftStore } from '@/store/drift';

interface Props { participantUuid: string; }

export function ServiceCardChips({ participantUuid }: Props) {
  const atoms = useGraphStore((s) =>
    s.nodes.filter((n) => n.parent_uuid === participantUuid && n.level === 'L4')
  );
  const driftedSet = useDriftStore((s) => s.driftedUuids);
  const rollupStaleSet = useRollupStore((s) => s.rollupStaleUuids);
  const untrackedSet = useRollupStore((s) => s.untrackedUuids);
  const substrateStates = useSubstrateStore((s) => s.nodeStates);

  if (atoms.length === 0) return null;

  return (
    <div className="flex flex-col gap-1 ml-2">
      {atoms.map((atom) => {
        const state = resolveNodeState(atom.uuid, driftedSet, rollupStaleSet, untrackedSet, substrateStates);
        return (
          <button
            key={atom.uuid}
            data-atom-uuid={atom.uuid}
            data-state={state}
            className={chipStyles({ state })}
            onClick={() => useGraphStore.getState().setSelectedNode(atom.uuid)}
          >
            {atom.name}
          </button>
        );
      })}
    </div>
  );
}

**Step 1 — `ServiceCard.tsx` — generic backend participant card.**

ServiceCard is the react-flow node component. Renders:
- Header: kind icon (small lucide icon for api/lib/data/external/job/cron/event) + name in monospace + atom-count chip
- Body (depends on kind):
  - For `api`: METHOD badge + monospace path (passes to EndpointCard sub-component for HTTP-specific layout)
  - For `lib`: monospace function signature header
  - For `data`: monospace `db.<table>.<op>` header
  - For `external`: SDK call header (`stripe.customers.update`)
  - For `cron`: schedule line (`cron: 0 * * * *`)
  - For `event`: event line (`event: stripe.customer.subscription.deleted`)
- Inputs section (parsed from `## Inputs`): syntax-highlighted JSON in a small fenced container
- Outputs section: 200/401/500 variants if multiple
- Side effects: bulleted list (each item rendered in monospace if it looks like code, else plain text)
- Right-side ServiceCardChips column

```tsx
import { memo, useMemo } from 'react';
import type { NodeProps } from '@xyflow/react';
import { Handle, Position } from '@xyflow/react';
import { cardKindStyles, methodBadgeStyles } from './cardStyles';
import { ServiceCardChips } from './ServiceCardChips';
import { resolveNodeState } from './contractNodeStyles';
import { parseBackendSections } from '@/lib/backendFrontmatter';
import { useDriftStore } from '@/store/drift';
import { useRollupStore } from '@/store/rollup';
import { useSubstrateStore } from '@/store/substrate';

export interface ServiceCardData extends Record<string, unknown> {
  uuid: string;
  kind: 'api' | 'lib' | 'data' | 'external' | 'job' | 'cron' | 'event';
  name: string;
  body: string;     // contract body markdown (passed in by GraphCanvasInner)
  method?: 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE';
  path?: string;
  schedule?: string;
  eventType?: string;
}

function ServiceCardImpl({ data }: NodeProps) {
  const d = data as ServiceCardData;
  const sections = useMemo(() => parseBackendSections(d.body || ''), [d.body]);
  const drifted = useDriftStore((s) => s.driftedUuids);
  const rollupStale = useRollupStore((s) => s.rollupStaleUuids);
  const untracked = useRollupStore((s) => s.untrackedUuids);
  const substrate = useSubstrateStore((s) => s.nodeStates);
  const state = resolveNodeState(d.uuid, drifted, rollupStale, untracked, substrate);

  return (
    <div className="flex">
      <div className={cardKindStyles({ kind: d.kind, state })} style={{ minWidth: 320, maxWidth: 480 }}>
        <Handle type="target" position={Position.Top} />
        <header className="flex items-center gap-2 px-3 py-2 border-b border-border/40">
          {/* method-specific header — see EndpointCard for api kind */}
          {d.kind === 'api' && d.method && d.path && (
            <>
              <span className={methodBadgeStyles({ method: d.method })}>{d.method}</span>
              <code className="text-sm font-mono">{d.path}</code>
            </>
          )}
          {d.kind === 'lib' && <code className="text-sm font-mono">{d.name}()</code>}
          {d.kind === 'data' && <code className="text-sm font-mono">{d.name}</code>}
          {d.kind === 'external' && <code className="text-sm font-mono">{d.name}</code>}
          {d.kind === 'cron' && <code className="text-xs font-mono">cron: {d.schedule}</code>}
          {d.kind === 'event' && <code className="text-xs font-mono">event: {d.eventType}</code>}
          {d.kind === 'job' && <code className="text-sm font-mono">{d.name}</code>}
        </header>

        <div className="px-3 py-2 space-y-2 text-xs">
          {sections.inputs && (
            <div>
              <div className="text-[10px] uppercase text-muted-foreground mb-0.5">Request</div>
              <pre className="rounded bg-muted/50 p-1.5 font-mono text-[11px] overflow-x-auto">
                {sections.inputs.schema}
              </pre>
            </div>
          )}
          {sections.outputs.length > 0 && (
            <div>
              <div className="text-[10px] uppercase text-muted-foreground mb-0.5">Responses</div>
              {sections.outputs.map((out, i) => (
                <pre key={i} className="rounded bg-muted/50 p-1.5 font-mono text-[11px] overflow-x-auto">
                  {out.schema}
                </pre>
              ))}
            </div>
          )}
          {sections.sideEffects.length > 0 && (
            <div>
              <div className="text-[10px] uppercase text-muted-foreground mb-0.5">Side effects</div>
              <ul className="list-disc list-inside space-y-0.5">
                {sections.sideEffects.map((eff, i) => (
                  <li key={i} className="font-mono text-[11px]">{eff}</li>
                ))}
              </ul>
            </div>
          )}
          {!sections.inputs && sections.outputs.length === 0 && sections.sideEffects.length === 0 && (
            <div className="text-muted-foreground italic">No schema declared (BACKEND-FM-01 not populated)</div>
          )}
        </div>
        <Handle type="source" position={Position.Bottom} />
      </div>
      <ServiceCardChips participantUuid={d.uuid} />
    </div>
  );
}

export const ServiceCard = memo(ServiceCardImpl);
```

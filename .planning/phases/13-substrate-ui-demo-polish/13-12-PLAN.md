<objective>
Replace the current ScreenCard live-iframe-on-canvas behavior with a two-mode design that fixes the "scrolling the page zooms the canvas" gesture conflict AND delivers a Chrome DevTools-grade fullscreen inspect mode for atom-to-contract resolution.

**Mode 1 (canvas, default).** ScreenCard renders a cached screenshot (`<img>`) with the existing AtomChipOverlay drawing chips at their captured rects. No live iframe is visible on canvas at any time. Wheel/scroll events flow to react-flow uninterrupted. A hidden iframe (offscreen at full target dimensions) loads the page once, fires capture-snapshot via the responder, stores `{dataUrl, rects, capturedAt}` keyed by URL, then stays hidden. Hover-only header buttons: ⤢ to expand, ↻ to re-snapshot.

**Mode 2 (fullscreen inspect).** Click ⤢ (or ⌘.) → ScreenViewerOverlay slides up over the Graph area only (Sidebar, bottom Inspector slot, and Chat panel remain visible). The hidden iframe is positioned via CSS transform into the overlay's iframe slot — no DOM reattachment, so scroll/form state survives. Toolbar: URL · Inspect toggle (⌘⇧C, defaults off) · ↻ · ⤢-exit. With inspect ON, cursor crosshair; mousemove fires responder walk-up to nearest `[data-contract-uuid]`; parent renders accent outline + cursor badge. Click an atom → `useGraphStore.setSelectedNodeUuid(uuid)` → the **existing** bottom Inspector slot at `AppShell.tsx:472-480` auto-renders ContractTab via the existing selection→inspector wiring at `AppShell.tsx:421-430`. Click outside an atom → deselect → Inspector auto-collapses. Esc layered: inspect-on → inspect-off → close overlay → no-op.

**Why no new contract panel.** The bottom Inspector slot in `AppShell.tsx` already does everything we need: drag-resize via `<ResizablePanel collapsible>`, auto-expand on selection, auto-collapse on deselect, ContractTab rendering. Building a separate "bottom-docked panel" would duplicate it. We sync inspect-mode selection into the existing graph store and let the existing Inspector react. Saves ~150 LOC and avoids two contract surfaces fighting for attention.

**Demo value.** Ties to ROADMAP SC 1 (ScreenCard with iframe) + SC 6 (substrate-state coloring across new node types) by upgrading ScreenCard's interaction model without changing its data shape. The DevTools-grade inspect feel reinforces the product thesis from VISION.md: contracts are the verification surface for atoms, and atoms are physical things you can point at on a real page.
</objective>

**Step 2 — Implement `handleCaptureSnapshot(requestId)`.**

Use SVG foreignObject + canvas. Same pattern as the parent-side `iframeScreenshot.ts` but operating on `document.documentElement` directly (no cross-origin barrier here):

```js
async function handleCaptureSnapshot(requestId) {
  try {
    const w = document.documentElement.scrollWidth;
    const h = document.documentElement.scrollHeight;
    const serialized = new XMLSerializer().serializeToString(document.documentElement);
    const svg = `<svg xmlns="http://www.w3.org/2000/svg" width="${w}" height="${h}"><foreignObject width="100%" height="100%">${serialized}</foreignObject></svg>`;
    const blob = new Blob([svg], { type: 'image/svg+xml;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const img = new Image();
    img.src = url;
    await new Promise((res, rej) => { img.onload = res; img.onerror = rej; });
    const canvas = document.createElement('canvas');
    canvas.width = w;
    canvas.height = h;
    canvas.getContext('2d').drawImage(img, 0, 0);
    URL.revokeObjectURL(url);
    const dataUrl = canvas.toDataURL('image/png');
    // Also collect current chip rects so screenshot+rects are temporally consistent.
    const rects = collectAllAtomRects();
    window.parent.postMessage({
      type: 'contract-ide:snapshot',
      requestId,
      dataUrl,
      rects,
    }, '*');
  } catch (e) {
    window.parent.postMessage({
      type: 'contract-ide:snapshot',
      requestId,
      error: String(e?.message || e),
    }, '*');
  }
}

function collectAllAtomRects() {
  const els = document.querySelectorAll('[data-contract-uuid]');
  const out = [];
  els.forEach((el) => {
    const r = el.getBoundingClientRect();
    out.push({
      uuid: el.dataset.contractUuid,
      rect: { x: r.x, y: r.y, width: r.width, height: r.height },
    });
  });
  return out;
}
```

**Step 3 — Add inspect-mode message handlers + DOM listeners.**

Two new message types from parent: `contract-ide:inspect-enable` and `contract-ide:inspect-disable`. When enabled, attach mousemove + click + cursor:crosshair on `document.documentElement`. When disabled, detach + reset cursor.

```js
let inspectActive = false;
let lastHoverUuid = null;

function handleInspectEnable() {
  if (inspectActive) return;
  inspectActive = true;
  document.documentElement.style.cursor = 'crosshair';
  document.addEventListener('mousemove', onInspectMouseMove, true);
  document.addEventListener('click', onInspectClick, true);
}

function handleInspectDisable() {
  if (!inspectActive) return;
  inspectActive = false;
  document.documentElement.style.cursor = '';
  document.removeEventListener('mousemove', onInspectMouseMove, true);
  document.removeEventListener('click', onInspectClick, true);
  lastHoverUuid = null;
  window.parent.postMessage({ type: 'contract-ide:inspect-hover', uuid: null }, '*');
}

function findAtomAncestor(el) {
  let cur = el;
  while (cur && cur !== document.body && cur !== document.documentElement) {
    if (cur.dataset?.contractUuid) return cur;
    cur = cur.parentElement;
  }
  return null;
}

function onInspectMouseMove(e) {
  const atomEl = findAtomAncestor(e.target);
  if (!atomEl) {
    if (lastHoverUuid !== null) {
      lastHoverUuid = null;
      window.parent.postMessage({ type: 'contract-ide:inspect-hover', uuid: null }, '*');
    }
    return;
  }
  const uuid = atomEl.dataset.contractUuid;
  if (uuid === lastHoverUuid) return;
  lastHoverUuid = uuid;
  const r = atomEl.getBoundingClientRect();
  window.parent.postMessage({
    type: 'contract-ide:inspect-hover',
    uuid,
    rect: { x: r.x, y: r.y, width: r.width, height: r.height },
  }, '*');
}

function onInspectClick(e) {
  const atomEl = findAtomAncestor(e.target);
  if (!atomEl) {
    // Click on empty region: deselect, but don't intercept the click — let the page handle it.
    window.parent.postMessage({ type: 'contract-ide:inspect-click', uuid: null }, '*');
    return;
  }
  // Click on atom: prevent navigation/form submission, send select.
  e.preventDefault();
  e.stopPropagation();
  const uuid = atomEl.dataset.contractUuid;
  const r = atomEl.getBoundingClientRect();
  window.parent.postMessage({
    type: 'contract-ide:inspect-click',
    uuid,
    rect: { x: r.x, y: r.y, width: r.width, height: r.height },
  }, '*');
}
```

**Step 1 — Extend `iframeChipPositioning.ts` with new message types.**

Add three new functions next to the existing `requestChipRects`:

```ts
export interface SnapshotResult {
  dataUrl: string;
  rects: Array<{ uuid: string; rect: { x: number; y: number; width: number; height: number } }>;
}

export async function requestSnapshot(
  iframe: HTMLIFrameElement,
  timeoutMs = 2000,
): Promise<SnapshotResult | null> {
  if (!iframe.contentWindow) return null;
  const requestId = crypto.randomUUID();
  return new Promise((resolve) => {
    const timer = setTimeout(() => {
      window.removeEventListener('message', onMsg);
      resolve(null);
    }, timeoutMs);
    function onMsg(e: MessageEvent) {
      if (e.source !== iframe.contentWindow) return;
      const msg = e.data;
      if (!msg || msg.type !== 'contract-ide:snapshot' || msg.requestId !== requestId) return;
      window.removeEventListener('message', onMsg);
      clearTimeout(timer);
      if (msg.error) { resolve(null); return; }
      resolve({ dataUrl: msg.dataUrl, rects: msg.rects ?? [] });
    }
    window.addEventListener('message', onMsg);
    iframe.contentWindow!.postMessage(
      { type: 'contract-ide:capture-snapshot', requestId },
      '*',
    );
  });
}

export function enableInspect(iframe: HTMLIFrameElement) {
  iframe.contentWindow?.postMessage({ type: 'contract-ide:inspect-enable' }, '*');
}

export function disableInspect(iframe: HTMLIFrameElement) {
  iframe.contentWindow?.postMessage({ type: 'contract-ide:inspect-disable' }, '*');
}

export type InspectHoverPayload =
  | { uuid: string; rect: { x: number; y: number; width: number; height: number } }
  | { uuid: null };
export type InspectClickPayload = InspectHoverPayload;

export function subscribeInspect(
  iframe: HTMLIFrameElement,
  onHover: (p: InspectHoverPayload) => void,
  onClick: (p: InspectClickPayload) => void,
): () => void {
  function handler(e: MessageEvent) {
    if (e.source !== iframe.contentWindow) return;
    const msg = e.data;
    if (!msg || typeof msg !== 'object') return;
    if (msg.type === 'contract-ide:inspect-hover') {
      onHover(msg.uuid === null ? { uuid: null } : { uuid: msg.uuid, rect: msg.rect });
    } else if (msg.type === 'contract-ide:inspect-click') {
      onClick(msg.uuid === null ? { uuid: null } : { uuid: msg.uuid, rect: msg.rect });
    }
  }
  window.addEventListener('message', handler);
  return () => window.removeEventListener('message', handler);
}
```

```tsx
        style={{

          top: isExpanded ? undefined : '-10000px',
          left: isExpanded ? undefined : '-10000px',
          width: '1280px',
          height: '800px',
          border: 'none',
          pointerEvents: isExpanded ? 'auto' : 'none',
          // ScreenViewerOverlay will override these via inline style when isExpanded
        }}
        data-screen-iframe-host={screenUuid}
      />
    )}
  </div>
);
```

**Step 1 — Build ScreenViewerOverlay.**

Mount as a sibling of the react-flow canvas inside `GraphCanvas.tsx`. The overlay covers the Graph area only by being absolutely-positioned inside the GraphCanvas container (which is the top portion of the center column per `AppShell.tsx:466-468`). The Sidebar, bottom Inspector, and Chat panel naturally remain visible and interactive because they're outside the GraphCanvas container.

```tsx
export function ScreenViewerOverlay() {
  const expandedUuid = useScreenViewerStore((s) => s.expandedScreenUuid);
  const close = useScreenViewerStore((s) => s.close);
  const inspectMode = useScreenViewerStore((s) => s.inspectMode);
  const toggleInspect = useScreenViewerStore((s) => s.toggleInspect);

  // Find the screen node in the graph store
  const screen = useGraphStore((s) =>
    expandedUuid ? s.nodes.find((n) => n.uuid === expandedUuid) : null
  );
  const fullUrl = screen ? buildFullUrl(screen) : null;

  // Iframe slot ref — measured each render to position the iframe via CSS transform
  const iframeSlotRef = useRef<HTMLDivElement>(null);

  // Find the iframe element by data attribute (mounted in ScreenCard)
  useLayoutEffect(() => {
    if (!expandedUuid) return;
    const iframe = document.querySelector<HTMLIFrameElement>(
      `iframe[data-screen-iframe-host="${expandedUuid}"]`
    );
    const slot = iframeSlotRef.current;
    if (!iframe || !slot) return;

    const positionIframe = () => {
      const slotRect = slot.getBoundingClientRect();
      // Iframe is rendered at 1280×800 — scale to fit the slot
      const scaleX = slotRect.width / 1280;
      const scaleY = slotRect.height / 800;
      const scale = Math.min(scaleX, scaleY);
      iframe.style.position = 'fixed';
      iframe.style.top = `${slotRect.top}px`;
      iframe.style.left = `${slotRect.left}px`;
      iframe.style.width = `${slotRect.width}px`;
      iframe.style.height = `${slotRect.height}px`;
      iframe.style.pointerEvents = 'auto';
      iframe.style.zIndex = '40'; // above overlay backdrop, below toolbar
      iframe.style.transform = ''; // we use width/height instead of transform-scale to avoid blurriness
    };

    positionIframe();
    const ro = new ResizeObserver(positionIframe);
    ro.observe(slot);
    window.addEventListener('resize', positionIframe);
    return () => {
      ro.disconnect();
      window.removeEventListener('resize', positionIframe);
      // Restore iframe to hidden offscreen position
      iframe.style.position = 'fixed';
      iframe.style.top = '-10000px';
      iframe.style.left = '-10000px';
      iframe.style.width = '1280px';
      iframe.style.height = '800px';
      iframe.style.pointerEvents = 'none';
      iframe.style.zIndex = '';
    };
  }, [expandedUuid]);

  if (!expandedUuid || !fullUrl) return null;

  return (
    <div
      className="absolute inset-0 z-30 flex flex-col bg-background/95 backdrop-blur-sm animate-in fade-in slide-in-from-bottom-4 duration-200"
      onClick={(e) => { if (e.target === e.currentTarget) close(); }}
    >
      {/* Toolbar */}
      <div className="flex items-center gap-2 px-4 py-2 border-b bg-background/80 z-50">
        <span className="text-xs text-muted-foreground font-mono">{fullUrl}</span>
        <div className="flex-1" />
        <button
          onClick={toggleInspect}
          className={`px-2 py-1 text-xs rounded ${inspectMode ? 'bg-accent text-accent-foreground' : 'hover:bg-muted'}`}
          title="Toggle inspect (⌘⇧C)"
        >
          Inspect
        </button>
        <button onClick={() => triggerRefresh(expandedUuid)} className="px-2 py-1 text-xs rounded hover:bg-muted" title="Refresh snapshot">↻</button>
        <button onClick={close} className="px-2 py-1 text-xs rounded hover:bg-muted" title="Exit (Esc)">⤢</button>
      </div>

      {/* Iframe slot — empty div, the iframe is positioned over it via fixed positioning */}
      <div ref={iframeSlotRef} className="flex-1 relative">
        {/* InspectHighlight renders here when inspectMode === true */}
        <InspectHighlight slotRef={iframeSlotRef} />
      </div>
    </div>
  );
}
```

**Step 3 — Why CSS transform / position:fixed instead of React Portal.**

Moving an iframe element across DOM trees (e.g., via `appendChild` or React Portal which uses appendChild internally) reattaches the iframe, which causes WebKit/Blink to reload the contentDocument — page state and scroll lost. The CSS-fixed approach keeps the iframe in its original DOM location (inside ScreenCard) but visually positions it over the overlay's slot. No reattachment, no reload.

**Step 4 — Animation.**

Use `animate-in fade-in slide-in-from-bottom-4 duration-200` (Tailwind animate utilities) for the overlay enter. Iframe positioning happens synchronously in `useLayoutEffect` — the iframe appears at its new position on the same frame as the overlay's first render, no janky offset.

```tsx
interface Props { slotRef: RefObject<HTMLDivElement> }

export function InspectHighlight({ slotRef }: Props) {
  const inspectMode = useScreenViewerStore((s) => s.inspectMode);
  const expandedUuid = useScreenViewerStore((s) => s.expandedScreenUuid);
  const hoverUuid = useScreenViewerStore((s) => s.hoverUuid);
  const hoverRect = useScreenViewerStore((s) => s.hoverRect);
  const setHover = useScreenViewerStore((s) => s.setHover);
  const selectedUuid = useGraphStore((s) => s.selectedNodeUuid);

  // Find the hovered atom node for badge content
  const hoverNode = useGraphStore((s) =>
    hoverUuid ? s.nodes.find((n) => n.uuid === hoverUuid) : null
  );

  useEffect(() => {
    if (!inspectMode || !expandedUuid) return;
    const iframe = document.querySelector<HTMLIFrameElement>(
      `iframe[data-screen-iframe-host="${expandedUuid}"]`
    );
    if (!iframe) return;

    enableInspect(iframe);
    const unsub = subscribeInspect(
      iframe,
      (p) => {
        if (p.uuid === null) {
          setHover(null, null);
        } else {
          setHover(p.uuid, p.rect);
        }
      },
      (p) => {
        // Click handling: sync to graph store
        useGraphStore.getState().setSelectedNodeUuid(p.uuid);
      },
    );
    return () => {
      unsub();
      disableInspect(iframe);
    };
  }, [inspectMode, expandedUuid, setHover]);

  if (!inspectMode) return null;

  // Translate iframe-local rect → slot-local rect.
  // The iframe is positioned to fill the slot (1:1 in slot coords), so we need
  // to scale by (slotWidth / iframeContentWidth). Iframe contentWidth = 1280
  // unless the page is wider; for v1 assume 1280.
  // Actually: since the iframe element is sized to the slot via CSS width/height,
  // the iframe's INTERNAL coordinate system (what the responder reports) is
  // 1280×800 but visually displayed at slot-size. So scale by:
  const slot = slotRef.current;
  if (!slot) return null;
  const slotRect = slot.getBoundingClientRect();
  const scaleX = slotRect.width / 1280;
  const scaleY = slotRect.height / 800;

  const renderRect = hoverRect ? {
    left: hoverRect.x * scaleX,
    top: hoverRect.y * scaleY,
    width: hoverRect.width * scaleX,
    height: hoverRect.height * scaleY,
  } : null;

  const isSelected = hoverUuid && hoverUuid === selectedUuid;

  return (
    <>
      {renderRect && (
        <div
          className={`absolute pointer-events-none border-2 transition-all duration-75 ${
            isSelected ? 'border-orange-500 bg-orange-500/10' : 'border-blue-500 bg-blue-500/5'
          }`}
          style={renderRect}
        />
      )}
      {hoverNode && renderRect && (
        <div
          className="absolute pointer-events-none bg-popover border rounded-md shadow-lg px-2 py-1 text-xs max-w-xs z-50"
          style={{
            left: renderRect.left,
            top: Math.max(0, renderRect.top - 32),
          }}
        >
          <div className="font-semibold">{hoverNode.name}</div>
          {hoverNode.intent && (
            <div className="text-muted-foreground line-clamp-1">{hoverNode.intent.split('\n')[0]}</div>
          )}
        </div>
      )}
    </>
  );
}
```

```ts
export function useScreenViewerHotkeys() {
  useEffect(() => {
    function onKey(e: KeyboardEvent) {
      const store = useScreenViewerStore.getState();
      const isMeta = e.metaKey || e.ctrlKey;

      // ⌘. → expand from currently-focused screen card
      if (isMeta && e.key === '.') {
        const focusedScreenUuid = getFocusedScreenUuid(); // helper that reads useGraphStore
        if (focusedScreenUuid) {
          e.preventDefault();
          store.expand(focusedScreenUuid);
        }
        return;
      }

      // ⌘⇧C → toggle inspect (only when overlay open)
      if (isMeta && e.shiftKey && e.key.toLowerCase() === 'c' && store.expandedScreenUuid) {
        e.preventDefault();
        store.toggleInspect();
        return;
      }

      // Esc → layered exit
      if (e.key === 'Escape' && store.expandedScreenUuid) {
        e.preventDefault();
        if (store.inspectMode) {
          store.setInspect(false);
        } else {
          store.close();
        }
        return;
      }
    }
    document.addEventListener('keydown', onKey);
    return () => document.removeEventListener('keydown', onKey);
  }, []);
}

function getFocusedScreenUuid(): string | null {
  // Read selectedNodeUuid; if it's a screen kind (UI L3), return it. Otherwise null.
  const sel = useGraphStore.getState().selectedNodeUuid;
  if (!sel) return null;
  const node = useGraphStore.getState().nodes.find((n) => n.uuid === sel);
  if (!node) return null;
  if (node.level === 'L3' && node.kind === 'UI') return sel;
  return null;
}
```

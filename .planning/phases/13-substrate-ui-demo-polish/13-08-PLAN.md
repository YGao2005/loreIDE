**Step 1 — `src/lib/diffToNodeMapper.ts` — pure parser + mapper.**

```typescript
export interface DiffHunk {
  filePath: string;          // 'lib/account/beginAccountDeletion.ts'
  oldStart: number;          // line in old file
  oldLines: number;
  newStart: number;
  newLines: number;
  rawHunk: string;           // raw hunk text for display
}

/**
 * Parse a unified diff text into hunks.
 *
 * Accepts standard `git diff` output. Handles:
 *   diff --git a/path b/path
 *   --- a/path
 *   +++ b/path
 *   @@ -OLD,L +NEW,L @@
 *
 * Returns ALL hunks across ALL files in the diff.
 */
export function parseDiffHunks(diffText: string): DiffHunk[] {
  const hunks: DiffHunk[] = [];
  const lines = diffText.split('\n');
  let currentFile: string | null = null;

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];
    // File header (use `+++` line; format: `+++ b/path/to/file`)
    if (line.startsWith('+++ ')) {
      const match = line.match(/^\+\+\+ b\/(.+)$/);
      if (match) currentFile = match[1];
      continue;
    }
    // Hunk header
    const hunkMatch = line.match(/^@@ -(\d+)(?:,(\d+))? \+(\d+)(?:,(\d+))? @@/);
    if (hunkMatch && currentFile) {
      const oldStart = parseInt(hunkMatch[1], 10);
      const oldLines = hunkMatch[2] ? parseInt(hunkMatch[2], 10) : 1;
      const newStart = parseInt(hunkMatch[3], 10);
      const newLines = hunkMatch[4] ? parseInt(hunkMatch[4], 10) : 1;

      // Capture raw hunk body until next @@ or +++
      const rawLines: string[] = [line];
      let j = i + 1;
      while (j < lines.length && !lines[j].startsWith('@@') && !lines[j].startsWith('+++ ') && !lines[j].startsWith('diff ')) {
        rawLines.push(lines[j]);
        j++;
      }
      hunks.push({
        filePath: currentFile,
        oldStart,
        oldLines,
        newStart,
        newLines,
        rawHunk: rawLines.join('\n'),
      });
    }
  }

  return hunks;
}

export interface NodeCodeRange {
  uuid: string;
  file: string;
  startLine?: number;
  endLine?: number;
}

/**
 * Map diff hunks to ContractNode uuids by file path + line range overlap.
 *
 * For each hunk, find all nodes whose code_ranges have:
 *   - file === hunk.filePath
 *   - line range overlaps hunk's NEW line range
 *
 * Returns Set of uuids (deduped).
 */
export function mapDiffToNodes(hunks: DiffHunk[], nodeRanges: NodeCodeRange[]): Set<string> {
  const affected = new Set<string>();
  for (const hunk of hunks) {
    const hunkStart = hunk.newStart;
    const hunkEnd = hunk.newStart + Math.max(hunk.newLines - 1, 0);
    for (const node of nodeRanges) {
      if (node.file !== hunk.filePath) continue;
      // If node has no line range, treat any hit on the file as affected (conservative)
      if (node.startLine == null || node.endLine == null) {
        affected.add(node.uuid);
        continue;
      }
      // Check overlap
      if (node.endLine >= hunkStart && node.startLine <= hunkEnd) {
        affected.add(node.uuid);
      }
    }
  }
  return affected;
}
```

**Step 3 — Rust IPC `analyze_pr_diff` in `commands/pr_review.rs`.**

```rust
use serde::{Deserialize, Serialize};
use sqlx::Row;
use tauri::State;
use tauri_plugin_sql::DbInstances;

#[derive(Debug, Serialize, Deserialize, Clone)]
pub struct PrReviewResult {
    pub affected_uuids: Vec<String>,
    pub intent_drifted_uuids: Vec<String>,  // subset of affected; substrate state is intent_drifted
    pub hunk_count: usize,
    pub file_count: usize,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct DiffHunkInput {
    pub file_path: String,
    pub new_start: i64,
    pub new_lines: i64,
}

/// Analyze a parsed diff (computed in TS) and return:
///   - affected_uuids: nodes whose code_ranges overlap a hunk
///   - intent_drifted_uuids: subset whose substrate state is 'intent_drifted'
///
/// The TS layer parses the diff and passes the structured hunks; Rust does the SQL join
/// (more efficient than fetching all nodes to TS).
#[tauri::command]
pub async fn analyze_pr_diff(
    db_instances: State<'_, DbInstances>,
    hunks: Vec<DiffHunkInput>,
) -> Result<PrReviewResult, String> {
    let instances = db_instances.0.read().await;
    let pool_ref = instances
        .get("sqlite:contract-ide.db")
        .ok_or_else(|| "DB not loaded".to_string())?;
    let pool = match pool_ref {
        tauri_plugin_sql::DbPool::Sqlite(p) => p,
        #[allow(unreachable_patterns)]
        _ => return Err("Unsupported pool".into()),
    };

    if hunks.is_empty() {
        return Ok(PrReviewResult {
            affected_uuids: vec![],
            intent_drifted_uuids: vec![],
            hunk_count: 0,
            file_count: 0,
        });
    }

    let file_count = hunks.iter().map(|h| &h.file_path).collect::<std::collections::HashSet<_>>().len();

    // Build query: find nodes where code_ranges JSON contains any of the hunk file paths
    // AND line ranges overlap. Simple approach: SELECT all nodes, parse code_ranges JSON in Rust, filter.
    // At hackathon scale (~50-200 nodes), this is fine.
    let rows = sqlx::query("SELECT uuid, code_ranges FROM nodes WHERE code_ranges IS NOT NULL")
        .fetch_all(pool)
        .await
        .map_err(|e| e.to_string())?;

    let mut affected = std::collections::HashSet::<String>::new();
    for row in rows {
        let uuid: String = row.get("uuid");
        let code_ranges_str: Option<String> = row.try_get("code_ranges").ok();
        let Some(cr) = code_ranges_str else { continue; };
        let parsed: Result<Vec<serde_json::Value>, _> = serde_json::from_str(&cr);
        let Ok(ranges) = parsed else { continue; };
        for r in ranges {
            let file = r.get("file").and_then(|v| v.as_str()).unwrap_or("");
            let start = r.get("startLine").and_then(|v| v.as_i64()).unwrap_or(0);
            let end = r.get("endLine").and_then(|v| v.as_i64()).unwrap_or(i64::MAX);
            for h in &hunks {
                if h.file_path == file {
                    let h_end = h.new_start + h.new_lines.saturating_sub(1).max(0);
                    if end >= h.new_start && start <= h_end {
                        affected.insert(uuid.clone());
                        break;
                    }
                    // Fallback: file matches but no line range — count as affected
                    if start == 0 && end == i64::MAX {
                        affected.insert(uuid.clone());
                        break;
                    }
                }
            }
        }
    }

    // Intersect with substrate_nodes whose state = 'intent_drifted'
    let mut intent_drifted = vec![];
    let exists_check: (i64,) = sqlx::query_as(
        "SELECT COUNT(*) FROM sqlite_master WHERE type='table' AND name='substrate_nodes'"
    )
    .fetch_one(pool)
    .await
    .map_err(|e| e.to_string())?;
    if exists_check.0 > 0 {
        let drifted_rows = sqlx::query("SELECT uuid FROM substrate_nodes WHERE state = 'intent_drifted'")
            .fetch_all(pool)
            .await
            .map_err(|e| e.to_string())?;
        for r in drifted_rows {
            let u: String = r.get("uuid");
            if affected.contains(&u) {
                intent_drifted.push(u);
            }
        }
    }

    Ok(PrReviewResult {
        affected_uuids: affected.into_iter().collect(),
        intent_drifted_uuids: intent_drifted,
        hunk_count: hunks.len(),
        file_count,
    })
}
```

```tsx
import { useState, useRef } from 'react';
import { invoke } from '@tauri-apps/api/core';
import { parseDiffHunks } from '@/lib/diffToNodeMapper';
import { useSubstrateStore } from '@/store/substrate';
import { PRReviewExplanation } from './PRReviewExplanation';

interface PrReviewResult {
  affected_uuids: string[];
  intent_drifted_uuids: string[];
  hunk_count: number;
  file_count: number;
}

export function PRReviewPanel({ open, onClose }: { open: boolean; onClose: () => void }) {
  const [diffText, setDiffText] = useState('');
  const [analyzing, setAnalyzing] = useState(false);
  const [result, setResult] = useState<PrReviewResult | null>(null);
  const previousStatesRef = useRef<Map<string, string | null>>(new Map());

  if (!open) return null;

  const onAnalyze = async () => {
    setAnalyzing(true);
    try {
      const hunks = parseDiffHunks(diffText);
      if (hunks.length === 0) {
        setResult({ affected_uuids: [], intent_drifted_uuids: [], hunk_count: 0, file_count: 0 });
        setAnalyzing(false);
        return;
      }
      const inputHunks = hunks.map((h) => ({
        file_path: h.filePath,
        new_start: h.newStart,
        new_lines: h.newLines,
      }));
      const r = await invoke<PrReviewResult>('analyze_pr_diff', { hunks: inputHunks });
      setResult(r);

      // Snapshot previous substrate states so cancel can restore them
      const sub = useSubstrateStore.getState();
      previousStatesRef.current = new Map();
      for (const uuid of r.affected_uuids) {
        previousStatesRef.current.set(uuid, sub.nodeStates.get(uuid) ?? null);
      }

      // Apply intent_drifted overlay to all affected (in real-world this would be more nuanced
      // but for demo simplicity: all affected get amber-orange treatment)
      sub.bulkSet(r.affected_uuids.map((uuid) => ({ uuid, state: 'intent_drifted' as const })));
    } catch (err) {
      console.error('[PRReviewPanel] analyze failed:', err);
    } finally {
      setAnalyzing(false);
    }
  };

  const onCancel = () => {
    // Restore previous substrate states
    const sub = useSubstrateStore.getState();
    for (const [uuid, prevState] of previousStatesRef.current) {
      if (prevState === null) {
        sub.clearNodeState(uuid);
      } else {
        sub.setNodeState(uuid, prevState as any);
      }
    }
    previousStatesRef.current.clear();
    setDiffText('');
    setResult(null);
    onClose();
  };

  return (
    <div
      className="fixed right-0 top-0 bottom-0 w-[420px] bg-background border-l border-border/40 shadow-xl z-40 flex flex-col"
      data-pr-review-panel
    >
      <header className="flex items-center justify-between px-3 py-2 border-b border-border/40">
        <h3 className="text-sm font-medium">PR Review — Intent Drift</h3>
        <button onClick={onCancel} className="text-xs text-muted-foreground hover:text-foreground">
          ✕ Close
        </button>
      </header>
      <div className="flex-1 overflow-y-auto p-3 space-y-3">
        <div>
          <label className="text-xs text-muted-foreground block mb-1">Paste raw diff (output of <code className="font-mono text-[11px]">git diff</code>)</label>
          <textarea
            value={diffText}
            onChange={(e) => setDiffText(e.target.value)}
            placeholder="diff --git a/path b/path&#10;..."
            className="w-full h-40 rounded border border-border/50 bg-muted/30 p-2 text-[11px] font-mono"
          />
        </div>
        <button
          onClick={onAnalyze}
          disabled={analyzing || !diffText.trim()}
          className="w-full rounded bg-primary text-primary-foreground py-1.5 text-sm font-medium disabled:opacity-50"
        >
          {analyzing ? 'Analyzing...' : 'Analyze diff'}
        </button>
        {result && (
          <div className="text-xs text-muted-foreground">
            {result.hunk_count} hunks across {result.file_count} files affecting {result.affected_uuids.length} atoms
            {result.intent_drifted_uuids.length > 0 && (
              <span className="text-orange-300"> ({result.intent_drifted_uuids.length} intent-drifted)</span>
            )}
          </div>
        )}
        {result && result.affected_uuids.length > 0 && (
          <PRReviewExplanation result={result} />
        )}
      </div>
    </div>
  );
}
```

```tsx
import { useGraphStore } from '@/store/graph';
import { SubstrateCitation } from '@/components/inspector/SubstrateCitation';

interface Props {
  result: { affected_uuids: string[]; intent_drifted_uuids: string[] };
}

export function PRReviewExplanation({ result }: Props) {
  const allNodes = useGraphStore((s) => s.nodes);
  const affectedNodes = allNodes.filter((n) => result.affected_uuids.includes(n.uuid));

  // Group by parent (participant surface)
  const byParent = new Map<string, typeof allNodes>();
  for (const n of affectedNodes) {
    const key = n.parent_uuid ?? '_root';
    if (!byParent.has(key)) byParent.set(key, []);
    byParent.get(key)!.push(n);
  }

  return (
    <div className="space-y-3">
      <h4 className="text-xs uppercase text-muted-foreground">Affected by intent</h4>
      {Array.from(byParent.entries()).map(([parentUuid, nodes]) => {
        const parent = allNodes.find((n) => n.uuid === parentUuid);
        return (
          <div key={parentUuid} className="space-y-1">
            <div className="text-sm font-medium">{parent?.name ?? parentUuid}</div>
            <ul className="space-y-1 ml-3">
              {nodes.map((n) => {
                const isDrifted = result.intent_drifted_uuids.includes(n.uuid);
                return (
                  <li key={n.uuid} className="text-xs text-muted-foreground flex items-center gap-1">
                    {isDrifted && <span className="text-orange-300">⚠</span>}

                      onClick={() => useGraphStore.getState().setSelectedNode(n.uuid)}
```

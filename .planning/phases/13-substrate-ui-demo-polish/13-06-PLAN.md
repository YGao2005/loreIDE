**Step 1 — `src/lib/flowChainAssembler.ts` — pure assembler function.**

Takes flow contract's members array + the array of all known contract nodes; returns react-flow nodes (with `screenCard` or `serviceCard` type) and edges (with CallShapeEdge type) ready to feed react-flow.

```typescript
import type { Node, Edge } from '@xyflow/react';
import { parseBackendSections } from './backendFrontmatter';
import type { ContractNode } from '@/ipc/types';

export interface FlowChainNode {
  uuid: string;
  name: string;
  kind: 'ui' | 'api' | 'lib' | 'data' | 'external' | 'job' | 'cron' | 'event';
  body: string;
  route?: string;          // for kind: 'ui'
  method?: 'GET' | 'POST' | 'PUT' | 'PATCH' | 'DELETE';
  path?: string;           // for kind: 'api'
  schedule?: string;       // for kind: 'cron'
  eventType?: string;      // for kind: 'event'
  isFocused?: boolean;     // true for the focused flow's trigger; controls live iframe vs screenshot
}

const VERTICAL_GAP = 120;     // px between consecutive participants
const CARD_X = 0;             // x-offset for cards in chain (centered later by react-flow fitView)

export function assembleFlowChain(
  flowMemberUuids: string[],
  allNodes: ContractNode[],
  focusedFlowUuid: string | null,    // for screenshot vs iframe decision
  thisFlowUuid: string,
): { nodes: Node[]; edges: Edge[] } {
  const nodeMap = new Map(allNodes.map((n) => [n.uuid, n]));
  const isThisFocused = focusedFlowUuid === thisFlowUuid;
  const result: Node[] = [];
  const edges: Edge[] = [];

  let y = 0;
  const ordered = flowMemberUuids
    .map((u) => nodeMap.get(u))
    .filter((n): n is ContractNode => Boolean(n));

  for (let i = 0; i < ordered.length; i++) {
    const node = ordered[i];
    const isTrigger = i === 0;
    const isUiTrigger = isTrigger && node.kind === 'ui';
    const reactFlowType = isUiTrigger ? 'screenCard' : 'serviceCard';

    const data: Record<string, unknown> = {
      uuid: node.uuid,
      name: node.name,
      kind: node.kind,
      body: node.contract_body ?? '',
      // Pass focus flag so ScreenCard renders live iframe only if focused
      isFocused: isThisFocused,
    };
    // Populate kind-specific fields by parsing body or contract metadata
    if (node.kind === 'ui') {
      data.route = (node as any).route ?? '';
    }
    if (node.kind === 'api') {
      // Method+path is encoded in name (e.g. "POST /accounts/:id/delete")
      const m = node.name.match(/^(GET|POST|PUT|PATCH|DELETE)\s+(\S+)/);
      if (m) {
        data.method = m[1];
        data.path = m[2];
      }
    }
    // ... cron / event etc.

    result.push({
      id: node.uuid,
      type: reactFlowType,
      position: { x: CARD_X, y },
      data,
      // Disable dragging in chain mode — layout is deterministic
      draggable: false,
    });

    if (i > 0) {
      const prev = ordered[i - 1];
      edges.push({
        id: `${prev.uuid}->${node.uuid}`,
        source: prev.uuid,
        target: node.uuid,
        type: 'callShape',
        data: deriveCallShape(prev.contract_body ?? '', node.contract_body ?? ''),
      });
    }

    // Estimate vertical step: ScreenCard is ~440px tall; ServiceCard ~150-220px
    const estimatedHeight = reactFlowType === 'screenCard' ? 440 : 200;
    y += estimatedHeight + VERTICAL_GAP;
  }

  return { nodes: result, edges };
}

export interface CallShape {
  label: string;          // e.g. "{ userId } → { userId, deletedAt }"  or "?"
  prevOutputUuid?: string; // for highlight wiring
  nextInputUuid?: string;
  matched: boolean;       // false if mismatched / unmappable
}

export function deriveCallShape(prevBody: string, nextBody: string): CallShape {
  const prev = parseBackendSections(prevBody);
  const next = parseBackendSections(nextBody);

  const prevOut = prev.outputs[0]?.schema ?? prev.outputs[0]?.raw;
  const nextIn = next.inputs?.schema ?? next.inputs?.raw;

  if (!prevOut || !nextIn) {
    return { label: '?', matched: false };
  }

  // Try parse JSON — if both are JSON, show key intersection
  try {
    const prevObj = JSON.parse(prevOut);
    const nextObj = JSON.parse(nextIn);
    const prevKeys = Object.keys(prevObj);
    const nextKeys = Object.keys(nextObj);
    const shared = prevKeys.filter((k) => nextKeys.includes(k));
    if (shared.length === 0) return { label: '?', matched: false };
    return { label: `{ ${shared.join(', ')} }`, matched: true };
  } catch {
    // Non-JSON or parse failed — show truncated raw
    const truncated = (prevOut + ' → ' + nextIn).slice(0, 64);
    return { label: truncated, matched: false };
  }
}
```

**Why this shape:** Per CHAIN-02 spec, "Mismatched / unmappable schemas render as `?` or are omitted; never render garbage." The `matched: false` flag drives a CSS class on the edge label so it visually reads as "we tried but couldn't" vs successful mapping.

import { memo } from 'react';
import { BaseEdge, EdgeLabelRenderer, getSmoothStepPath, type EdgeProps } from '@xyflow/react';
import type { CallShape } from '@/lib/flowChainAssembler';

function CallShapeEdgeImpl({ id, sourceX, sourceY, targetX, targetY, data }: EdgeProps) {
  const [edgePath, labelX, labelY] = getSmoothStepPath({
    sourceX,
    sourceY,
    targetX,
    targetY,
  });

  const shape = data as CallShape | undefined;
  const label = shape?.label ?? '';
  const muted = shape?.matched === false;

  return (
    <>
      <BaseEdge id={id} path={edgePath} style={{ strokeWidth: 1.5, stroke: 'rgba(148, 163, 184, 0.6)' }} />
      {label && (
        <EdgeLabelRenderer>
          <div
            className={`absolute pointer-events-none rounded px-2 py-0.5 text-[11px] font-mono ${
              muted ? 'bg-slate-800/80 text-slate-400' : 'bg-slate-800/90 text-slate-200'
            }`}
            style={{
              transform: `translate(-50%, -50%) translate(${labelX}px, ${labelY}px)`,
            }}
          >
            {label}
          </div>
        </EdgeLabelRenderer>
      )}
    </>
  );
}

export const CallShapeEdge = memo(CallShapeEdgeImpl);

**Step 1 — `iframeScreenshot.ts` — capture mechanism.**

Two paths:

1. **html2canvas-style** (DOM rendering): expensive, hits cross-origin restrictions, slow.
2. **Tauri screenshot IPC**: native screenshot of an offscreen window or use `webview2-com`-equivalent on macOS. Cleaner.

For hackathon scope: implement the **same-origin canvas approach** (since the iframe is localhost):

```typescript
/**
 * Capture a screenshot of an iframe by drawing its contentDocument to a canvas.
 * Same-origin only (works for Phase 4 localhost CSP).
 *
 * Returns a data URL (PNG). null on failure.
 *
 * Performance budget per ROADMAP: ≤120ms capture.
 * If this proves slow, fallback to Tauri-side capture_route_screenshot IPC.
 */
export async function captureIframeScreenshot(iframe: HTMLIFrameElement): Promise<string | null> {
  try {
    const doc = iframe.contentDocument;
    if (!doc) return null;
    // Approach: serialize the DOM to a foreignObject SVG, then rasterize via canvas.
    // This is the standard html2canvas trick without the dep weight.
    const w = iframe.clientWidth;
    const h = iframe.clientHeight;
    const xml = new XMLSerializer().serializeToString(doc.documentElement);
    const svg = `<svg xmlns="http://www.w3.org/2000/svg" width="${w}" height="${h}">
      <foreignObject width="100%" height="100%">${xml}</foreignObject>
    </svg>`;
    const blob = new Blob([svg], { type: 'image/svg+xml;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const img = new Image();
    img.crossOrigin = 'anonymous';
    return await new Promise<string | null>((resolve) => {
      img.onload = () => {
        const canvas = document.createElement('canvas');
        canvas.width = w;
        canvas.height = h;
        const ctx = canvas.getContext('2d');
        if (!ctx) return resolve(null);
        ctx.drawImage(img, 0, 0, w, h);
        URL.revokeObjectURL(url);
        try {
          const dataUrl = canvas.toDataURL('image/png');
          resolve(dataUrl);
        } catch (err) {
          // Tainted canvas (cross-origin): can't toDataURL
          console.warn('[iframeScreenshot] canvas tainted; falling back to Tauri IPC:', err);
          resolve(null);
        }
      };
      img.onerror = () => {
        URL.revokeObjectURL(url);
        resolve(null);
      };
      img.src = url;
    });
  } catch (err) {
    console.error('[iframeScreenshot] capture failed:', err);
    return null;
  }
}
```

export function FlowChainLayout() {

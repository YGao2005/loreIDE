**Step 1 — `src/store/citation.ts` — citation halo store.**

```typescript
import { create } from 'zustand';

interface CitationState {
  // For halo wiring: when a citation is clicked, this uuid lights up across the chain for ~2s
  highlightedUuid: string | null;
  // For modal: which citation is currently displayed
  openCitationUuid: string | null;
  highlight: (uuid: string, durationMs?: number) => void;
  openCitation: (uuid: string) => void;
  closeCitation: () => void;
  clearHighlight: () => void;
}

export const useCitationStore = create<CitationState>((set) => {
  let timeoutHandle: ReturnType<typeof setTimeout> | null = null;
  return {
    highlightedUuid: null,
    openCitationUuid: null,
    highlight: (uuid, durationMs = 2000) => {
      if (timeoutHandle) clearTimeout(timeoutHandle);
      set({ highlightedUuid: uuid });
      timeoutHandle = setTimeout(() => {
        set({ highlightedUuid: null });
        timeoutHandle = null;
      }, durationMs);
    },
    openCitation: (uuid) => set({ openCitationUuid: uuid }),
    closeCitation: () => set({ openCitationUuid: null }),
    clearHighlight: () => {
      if (timeoutHandle) clearTimeout(timeoutHandle);
      set({ highlightedUuid: null });
      timeoutHandle = null;
    },
  };
});
```

**Step 2 — `SubstrateCitation.tsx` — inline pill component.**

```tsx
import { memo } from 'react';
import { useCitationStore } from '@/store/citation';

interface Props {
  uuid: string;
  shortLabel?: string;  // e.g. 'dec-soft-delete-30day-grace-2026-02-18' or shorter
  haloOnClick?: boolean; // when true (Beat 3 sidebar usage), also calls highlight to ripple chain
}

function SubstrateCitationImpl({ uuid, shortLabel, haloOnClick = true }: Props) {
  const open = useCitationStore((s) => s.openCitation);
  const highlight = useCitationStore((s) => s.highlight);
  const label = shortLabel ?? uuid.slice(0, 16);

  return (
    <button
      onClick={(e) => {
        e.stopPropagation();
        open(uuid);
        if (haloOnClick) highlight(uuid);
      }}
      className="inline-flex items-center gap-0.5 rounded-md border border-blue-500/30 bg-blue-500/10 px-1.5 py-0.5 text-[10px] font-mono text-blue-300 hover:border-blue-400/60 hover:bg-blue-500/20 transition-colors"
    >
      [{label}]
    </button>
  );
}

export const SubstrateCitation = memo(SubstrateCitationImpl);
```

**Step 3 — `SourceArchaeologyModal.tsx` — verbatim quote display.**

Use the existing shadcn Dialog primitive (Phase 7 Plan 07-04 wired this).

```tsx
import { useEffect, useState } from 'react';
import { Dialog, DialogContent, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { useCitationStore } from '@/store/citation';
import { getSubstrateNodeDetail, type SubstrateNodeSummary } from '@/ipc/substrate';

export function SourceArchaeologyModal() {
  const openUuid = useCitationStore((s) => s.openCitationUuid);
  const close = useCitationStore((s) => s.closeCitation);
  const [detail, setDetail] = useState<SubstrateNodeSummary | null>(null);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    if (!openUuid) {
      setDetail(null);
      return;
    }
    setLoading(true);
    getSubstrateNodeDetail(openUuid)
      .then(setDetail)
      .catch((err) => {
        console.error('[SourceArchaeologyModal] fetch failed:', err);
        setDetail(null);
      })
      .finally(() => setLoading(false));
  }, [openUuid]);

  return (
    <Dialog open={Boolean(openUuid)} onOpenChange={(o) => !o && close()}>
      <DialogContent className="max-w-2xl">
        <DialogHeader>
          <DialogTitle className="font-mono text-sm">{detail?.name ?? openUuid}</DialogTitle>
        </DialogHeader>

        {loading && <div className="text-sm text-muted-foreground">Loading...</div>}

        {detail && !loading && (
          <div className="space-y-3">
            <div className="flex items-center gap-2 text-xs text-muted-foreground">
              <span className="rounded bg-muted/50 px-2 py-0.5 font-mono">{detail.kind}</span>
              {detail.actor && <span>actor: {detail.actor}</span>}
              {detail.confidence && <span>confidence: {detail.confidence}</span>}
              {detail.state && (
                <span className="rounded bg-muted/50 px-2 py-0.5 font-mono">{detail.state}</span>
              )}
            </div>

            {detail.summary && (
              <p className="text-sm text-muted-foreground">{detail.summary}</p>
            )}

            {detail.verbatim_quote ? (
              <div className="rounded border border-border/40 bg-muted/30 p-3">
                <div className="mb-1 text-[10px] uppercase text-muted-foreground">Verbatim quote</div>
                <blockquote className="text-sm whitespace-pre-wrap font-mono leading-relaxed">
                  {detail.verbatim_quote}
                </blockquote>
              </div>
            ) : (
              <div className="rounded border border-amber-500/40 bg-amber-500/10 p-3 text-sm text-amber-200">
                No verbatim quote on this node — was it hand-seeded? (Pitfall 4 in 13-RESEARCH.md)
              </div>
            )}

            {(detail.session_id || detail.turn_ref) && (
              <div className="text-[11px] font-mono text-muted-foreground">
                source: {detail.session_id}{detail.turn_ref ? `:${detail.turn_ref}` : ''}
              </div>
            )}
          </div>
        )}

        {!detail && !loading && openUuid && (
          <div className="text-sm text-muted-foreground">No detail found for {openUuid}</div>
        )}
      </DialogContent>
    </Dialog>
  );
}
```

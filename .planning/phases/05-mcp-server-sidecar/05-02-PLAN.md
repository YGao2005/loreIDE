/** Decode a sqlite row (as returned by better-sqlite3's Statement.all/get) into
 *  the ContractNodeRow shape the tools emit. Handles the JSON TEXT columns
 *  and the i64->boolean coercion (better-sqlite3 returns 0/1, not true/false). */
export function decodeNodeRow(r: Record<string, unknown>): ContractNodeRow {
  const codeRangesRaw = typeof r.code_ranges === 'string' ? r.code_ranges : null;
  let codeRanges: CodeRange[] = [];
  if (codeRangesRaw) {
    try {
      codeRanges = JSON.parse(codeRangesRaw) as CodeRange[];
    } catch {
      codeRanges = [];
    }
  }
  return {
    uuid: r.uuid as string,
    level: r.level as string,
    name: r.name as string,
    kind: (r.kind as string | null) ?? 'unknown',
    code_ranges: codeRanges,
    parent_uuid: (r.parent_uuid as string | null) ?? null,
    is_canonical: (r.is_canonical as number) !== 0,
    code_hash: (r.code_hash as string | null) ?? null,
    contract_hash: (r.contract_hash as string | null) ?? null,
    human_pinned: (r.human_pinned as number) !== 0,
    route: (r.route as string | null) ?? null,
    derived_at: (r.derived_at as string | null) ?? null,
    contract_body: (r.contract_body as string | null) ?? null,
  };
}

import { getDb } from '../db';

export async function findByIntent(query: string, limit: number) {
  const db = getDb();
  // nodes_fts MATCH is FTS5's bm25-ranked search. Columns in the virtual table
  // come from Plan 01-02 migration v1; verify via `PRAGMA table_info(nodes_fts)`
  // on the dev DB if the schema differs from this query.
  const rows = db
    .prepare(
      `
      SELECT n.uuid, n.name, n.level, n.kind,
             snippet(nodes_fts, -1, '**', '**', '...', 20) AS snippet
      FROM nodes_fts
      JOIN nodes n ON n.uuid = nodes_fts.uuid
      WHERE nodes_fts MATCH ?
      ORDER BY rank
      LIMIT ?
    `,
    )
    .all(query, limit) as Array<{
      uuid: string;
      name: string;
      level: string;
      kind: string;
      snippet: string;
    }>;

  if (rows.length === 0) {
    return {
      content: [
        { type: 'text' as const, text: `No contracts found matching: ${query}` },
      ],
    };
  }

  const text = rows
    .map(
      (r) =>
        `UUID: ${r.uuid}\nName: ${r.name} (${r.level} ${r.kind})\n${r.snippet}`,
    )
    .join('\n---\n');

  return { content: [{ type: 'text' as const, text }] };
}

**F. `mcp-sidecar/src/tools/update_contract.ts`** — the single write path. Reads the existing sidecar, parses frontmatter + body, applies `frontmatter_patch` (shallow merge) and replaces body text, writes via temp file + atomic rename. This is the Node-side equivalent of Plan 02-02's Rust `write_contract`. MCP-03 single-writer invariant is upheld because: (a) the `better-sqlite3` connection is `readonly:true` so any SELECT/INSERT against SQLite would throw; (b) the only persistence here is the filesystem `.md` write; (c) Rust's watcher (Plan 02-03) picks up the change and runs `refresh_nodes` → SQLite update, so Rust remains the sole SQLite writer:

```typescript
import fs from 'node:fs';
import path from 'node:path';
import { getRepoPath } from '../db';
import type { ContractFrontmatter } from '../types';

interface UpdateContractArgs {
  uuid: string;
  body: string;
  frontmatter_patch?: Record<string, unknown>;
}

export async function updateContract({
  uuid,
  body,
  frontmatter_patch,
}: UpdateContractArgs) {
  let repoPath: string;
  try {
    repoPath = getRepoPath();
  } catch (e) {
    return {
      content: [
        {
          type: 'text' as const,
          text: `ERROR: ${e instanceof Error ? e.message : String(e)}`,
        },
      ],
    };
  }

  const sidecarPath = path.join(repoPath, '.contracts', `${uuid}.md`);
  if (!fs.existsSync(sidecarPath)) {
    return {
      content: [
        { type: 'text' as const, text: `ERROR: sidecar not found: ${sidecarPath}` },
      ],
    };
  }

  const original = fs.readFileSync(sidecarPath, 'utf-8');
  const { fm, originalBody } = parseSidecarMinimal(original);
  if (!fm) {
    return {
      content: [
        {
          type: 'text' as const,
          text: `ERROR: failed to parse existing sidecar frontmatter at ${sidecarPath}`,
        },
      ],
    };
  }

  // Apply patch shallowly; null values explicitly overwrite. Do NOT allow the
  // patch to change the UUID — identity is immutable (DATA-04).
  const patched: ContractFrontmatter = { ...fm, ...(frontmatter_patch ?? {}) } as ContractFrontmatter;
  patched.uuid = fm.uuid;

  const newContent = serializeSidecar(patched, body.length > 0 ? body : originalBody);

  const tmpPath = `${sidecarPath}.tmp-${process.pid}`;
  fs.writeFileSync(tmpPath, newContent, 'utf-8');
  fs.renameSync(tmpPath, sidecarPath); // atomic within the same filesystem

  return {
    content: [
      {
        type: 'text' as const,
        text:
          `Updated ${sidecarPath} — Rust watcher will propagate to SQLite within 2s. ` +
          `(Single-writer upheld: MCP sidecar never writes SQLite directly.)`,
      },
    ],
  };
}

/**
 * Minimal frontmatter parser for update_contract. Mirrors the behaviour of
 * contract-ide/src-tauri/src/sidecar/frontmatter.rs parse_sidecar but only
 * needs JSON-round-trip correctness for the fields this tool touches.
 *
 * A full YAML lib would be preferable but adds dependency weight. Since the
 * Rust writer produces a deterministic format (Plan 02-01), a line-based
 * parser handling ---\n…\n---\n fences + `key: value` + list syntax is
 * sufficient for Phase 5 round-trip. If Phase 6/8 needs richer patches,
 * revisit with a yaml dep.
 *
 * Parser chosen approach: split on the canonical "\n---\n" closing fence and
 * hand-parse the YAML block with a small deterministic walker. If the body
 * contains a literal "---" line, we still find the first closing fence
 * correctly because the opening "---\n" is skipped and we search from there.
 */
function parseSidecarMinimal(content: string): {
  fm: ContractFrontmatter | null;
  originalBody: string;
} {
  const bom = content.startsWith('\ufeff') ? 1 : 0;
  const rest = content.slice(bom).replace(/\r\n/g, '\n');
  if (!rest.startsWith('---\n')) {
    return { fm: null, originalBody: '' };
  }
  const after = rest.slice(4);
  const closeIdx = after.indexOf('\n---\n');
  if (closeIdx < 0) {
    return { fm: null, originalBody: '' };
  }
  const yamlBlock = after.slice(0, closeIdx);
  const body = after.slice(closeIdx + 5).replace(/^\n+/, '');
  const fm = parseYamlFrontmatter(yamlBlock);
  return { fm, originalBody: body };
}

/**
 * Hand parser for the deterministic frontmatter shape our Rust writer emits.
 * Accepts: scalar strings, numbers, booleans, `null`, and `- item` arrays
 * (neighbors, code_ranges entries). Does NOT support nested maps outside of
 * the code_ranges list of {file, start_line, end_line} objects.
 */
function parseYamlFrontmatter(yaml: string): ContractFrontmatter | null {
  const lines = yaml.split('\n');
  const fm: Record<string, unknown> = {
    format_version: 1,
    uuid: '',
    kind: '',
    level: '',
    parent: null,
    neighbors: [] as string[],
    code_ranges: [] as Array<{ file: string; start_line: number; end_line: number }>,
    code_hash: null,
    contract_hash: null,
    human_pinned: false,
    route: null,
    derived_at: null,
  };

  let i = 0;
  while (i < lines.length) {
    const line = lines[i];
    if (line.trim() === '') {
      i++;
      continue;
    }
    const match = /^([A-Za-z_][A-Za-z0-9_]*):\s*(.*)$/.exec(line);
    if (!match) {
      i++;
      continue;
    }
    const key = match[1];
    const rawVal = match[2];

    if (rawVal === '' || rawVal === undefined) {
      // Block structure follows. If next non-empty line is indented `- ` it's a list.
      const listItems: string[] = [];
      const rangeItems: Array<{ file: string; start_line: number; end_line: number }> = [];
      i++;
      while (i < lines.length && (lines[i].startsWith('  ') || lines[i].startsWith('- '))) {
        const l = lines[i];
        if (l.startsWith('- ')) {
          // Either a scalar list entry ("- uuid-string") or the start of a code_ranges object.
          const itemBody = l.slice(2);
          if (/^file:\s*/.test(itemBody)) {
            const rangeObj: Record<string, unknown> = {};
            const fileMatch = /^file:\s*(.*)$/.exec(itemBody);
            if (fileMatch) rangeObj.file = unquote(fileMatch[1]);
            i++;
            while (i < lines.length && lines[i].startsWith('    ')) {
              const sub = lines[i].trim();
              const subMatch = /^([A-Za-z_][A-Za-z0-9_]*):\s*(.*)$/.exec(sub);
              if (subMatch) rangeObj[subMatch[1]] = parseScalar(subMatch[2]);
              i++;
            }
            rangeItems.push(rangeObj as { file: string; start_line: number; end_line: number });
            continue;
          } else {
            listItems.push(unquote(itemBody));
            i++;
          }
        } else {
          i++;
        }
      }
      if (key === 'code_ranges') fm.code_ranges = rangeItems;
      else if (key === 'neighbors') fm.neighbors = listItems;
      continue;
    }

    fm[key] = parseScalar(rawVal);
    i++;
  }

  if (!fm.uuid) return null;
  return fm as unknown as ContractFrontmatter;
}

function parseScalar(v: string): unknown {
  const t = v.trim();
  if (t === 'null' || t === '~' || t === '') return null;
  if (t === 'true') return true;
  if (t === 'false') return false;
  if (/^-?\d+$/.test(t)) return Number(t);
  return unquote(t);
}

function unquote(s: string): string {
  const t = s.trim();
  if ((t.startsWith('"') && t.endsWith('"')) || (t.startsWith("'") && t.endsWith("'"))) {
    return t.slice(1, -1);
  }
  return t;
}

/**
 * Emit sidecar .md text. Keeps the field order Plan 02-01 locked
 * (format_version first, derived_at last) so a round-trip through
 * parse→write produces byte-equivalent frontmatter for tests.
 */
function serializeSidecar(fm: ContractFrontmatter, body: string): string {
  const lines: string[] = ['---'];
  lines.push(`format_version: ${fm.format_version}`);
  lines.push(`uuid: ${fm.uuid}`);
  lines.push(`kind: ${fm.kind}`);
  lines.push(`level: ${fm.level}`);
  lines.push(`parent: ${emitScalar(fm.parent)}`);
  lines.push(`neighbors:`);
  if (fm.neighbors.length === 0) {
    lines[lines.length - 1] = 'neighbors: []';
  } else {
    for (const n of fm.neighbors) lines.push(`  - ${n}`);
  }
  lines.push(`code_ranges:`);
  if (fm.code_ranges.length === 0) {
    lines[lines.length - 1] = 'code_ranges: []';
  } else {
    for (const r of fm.code_ranges) {
      lines.push(`  - file: ${r.file}`);
      lines.push(`    start_line: ${r.start_line}`);
      lines.push(`    end_line: ${r.end_line}`);
    }
  }
  lines.push(`code_hash: ${emitScalar(fm.code_hash)}`);
  lines.push(`contract_hash: ${emitScalar(fm.contract_hash)}`);
  lines.push(`human_pinned: ${fm.human_pinned ? 'true' : 'false'}`);
  lines.push(`route: ${emitScalar(fm.route)}`);
  lines.push(`derived_at: ${emitScalar(fm.derived_at)}`);
  lines.push('---');
  lines.push('');
  lines.push(body.replace(/^\n+/, ''));
  return lines.join('\n');
}

function emitScalar(v: unknown): string {
  if (v === null || v === undefined) return 'null';
  return String(v);
}
```

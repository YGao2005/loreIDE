    ```rust
    pub const DISTILLER_PROMPT: &str = r#"You are extracting reusable SUBSTRATE NODES from a software-development session transcript.

    A typed observation that should guide future work on this codebase. Substrate nodes are
    **reusable** — they would save time or prevent bugs if automatically surfaced to a future
    agent on a relevant task.

    ## Five node types

    - **constraint**: A rule that must hold (e.g., "Always canonicalize file paths from JS in Rust commands")
    - **decision**: A choice the team made and committed to (e.g., "We use soft-delete with 30-day grace")
    - **open_question**: A question raised but not resolved in this session (e.g., "Should we cache contract bodies in MCP?")
    - **resolved_question**: A question raised AND answered in this session (e.g., "Q: Use shadcn-Tabs? A: No, button strips")
    - **attempt**: A pattern that was tried but didn't fully work — useful as a "don't repeat this exactly" signal

    ## Output schema

    For each substrate node, emit ONE JSON object matching:

    {
      "type": "constraint" | "decision" | "open_question" | "resolved_question" | "attempt",
      "text": "imperative or declarative statement, ONE sentence",
      "scope": "global" | "module:<path-pattern>" | "task-pattern:<short>",
      "applies_when": "semantic trigger for retrieval — be specific enough to avoid false positives, broad enough to catch paraphrases",
      "source": {
        "quote": "verbatim quote from transcript justifying this node",
        "actor": "user" | "claude" | "derived"
      },
      "confidence": "explicit" | "inferred"
    }

    ## Quality bar

    Each substrate node should answer: "Would I want this injected automatically into a future
    session whose goal matches `applies_when`?" If no, drop it.

    For constraint and decision nodes, prefer fewer high-quality nodes over many low-quality ones.

    For open_question and resolved_question nodes, capture them whenever a question gets raised
    or settled — these are higher-volume but cheap to keep.

    ## Input transcript

    <transcript>
    {filtered_text}
    </transcript>

    Output: a JSON object {"nodes": [...]} containing all substrate nodes you extracted.
    "#;

    pub fn substrate_node_schema() -> serde_json::Value {
        serde_json::json!({
            "type": "object",
            "properties": {
                "nodes": {
                    "type": "array",
                    "items": {
                        "type": "object",
                        "required": ["type", "text", "applies_when", "source", "confidence"],
                        "properties": {
                            "type": { "type": "string", "enum": ["constraint","decision","open_question","resolved_question","attempt"] },
                            "text": { "type": "string", "minLength": 1 },
                            "scope": { "type": ["string", "null"] },
                            "applies_when": { "type": "string", "minLength": 1 },
                            "source": {
                                "type": "object",
                                "required": ["quote", "actor"],
                                "properties": {
                                    "quote": { "type": "string", "minLength": 1 },
                                    "actor": { "type": "string", "enum": ["user","claude","derived"] }
                                }
                            },
                            "confidence": { "type": "string", "enum": ["explicit","inferred"] }
                        }
                    }
                }
            },
            "required": ["nodes"]
        })
    }
    ```

    **Step 3 — Create `contract-ide/src-tauri/src/distiller/pipeline.rs`:**

    Implement `distill_episode(app, episode_id)` per Pattern 1 in 11-RESEARCH.md. Critical structure:

    ```rust
    use crate::distiller::{prompt, state::DistillerLocks, types::*};
    use sha2::{Digest, Sha256};
    use sqlx::SqlitePool;
    use tauri::{AppHandle, Emitter, Manager};
    use tauri_plugin_shell::ShellExt;
    use tokio::time::{timeout, Duration};

    pub async fn distill_episode(
        app: &AppHandle,
        episode_id: &str,
    ) -> Result<usize, String> {
        // 1. Load episode + session metadata
        let pool: tauri::State<crate::DbInstances> = app.state();
        // (Match the existing DbPool::Sqlite extraction pattern from commands/nodes.rs)
        let pool_guard = pool.0.read().map_err(|e| format!("pool lock: {e}"))?;
        let DbPool::Sqlite(sqlite_pool) = pool_guard.get("sqlite:contract-ide.sqlite").ok_or("no sqlite pool")? else {
            return Err("not sqlite".into());
        };
        let row: (String, String, i64, Option<String>) = sqlx::query_as(
            "SELECT e.session_id, e.filtered_text, e.start_line, s.cwd_key
             FROM episodes e
             LEFT JOIN sessions s ON s.session_id = e.session_id
             WHERE e.episode_id = ?",
        )
        .bind(episode_id)
        .fetch_one(sqlite_pool).await
        .map_err(|e| format!("episode lookup: {e}"))?;
        let (session_id, filtered_text, start_line, _cwd_key) = row;

        // 2. Acquire per-session lock (Pattern 1 sub-pattern 3)
        let locks = app.state::<DistillerLocks>();
        let session_guard = locks.for_session(&session_id);
        let _lock = session_guard.lock().await;

        // 3. Run claude -p with --json-schema, 60s timeout
        let schema = prompt::substrate_node_schema();
        let prompt_text = prompt::DISTILLER_PROMPT.replace("{filtered_text}", &filtered_text);

        let claude_future = app.shell()
            .command("claude")
            .args([
                "-p", &prompt_text,
                "--output-format", "json",
                "--json-schema", &schema.to_string(),
                "--bare",
            ])
            .output();

        let output = timeout(Duration::from_secs(60), claude_future).await
            .map_err(|_| {
                // timeout — write dead-letter
                tokio::spawn(write_dead_letter_owned(app.clone(), episode_id.to_string(), "timeout".to_string(), None));
                "distiller timeout".to_string()
            })?
            .map_err(|e| {
                tokio::spawn(write_dead_letter_owned(app.clone(), episode_id.to_string(), "claude_exit_nonzero".to_string(), Some(e.to_string())));
                format!("claude spawn: {e}")
            })?;

        if !output.status.success() {
            let stderr = String::from_utf8_lossy(&output.stderr).to_string();
            write_dead_letter(sqlite_pool, episode_id, "claude_exit_nonzero", &stderr).await?;
            return Err("distiller failed (claude non-zero exit)".into());
        }

        // 4. Parse outer JSON, drill into structured_output.nodes (Pitfall 1)
        let response: serde_json::Value = match serde_json::from_slice(&output.stdout) {
            Ok(v) => v,
            Err(e) => {
                write_dead_letter(sqlite_pool, episode_id, "json_parse", &String::from_utf8_lossy(&output.stdout)).await?;
                return Err(format!("response parse: {e}"));
            }
        };

        let structured = response.get("structured_output");
        let nodes_raw = structured
            .and_then(|v| v.get("nodes"))
            .and_then(|v| v.as_array());

        let nodes_raw = match nodes_raw {
            Some(arr) => arr,
            None => {
                let truncated = response.to_string();
                let trimmed = &truncated[..truncated.len().min(4096)];
                write_dead_letter(sqlite_pool, episode_id, "schema_mismatch", trimmed).await?;
                return Err("missing structured_output.nodes".into());
            }
        };

        // 5. Idempotent upsert — stable UUID per (session_id, start_line, text-prefix-120)
        let mut upserted = 0usize;
        let now = chrono::Utc::now().to_rfc3339();
        for raw in nodes_raw {
            let text = raw.get("text").and_then(|v| v.as_str()).unwrap_or("");
            if text.is_empty() {
                continue;
            }
            let node_type = raw.get("type").and_then(|v| v.as_str()).unwrap_or("constraint");

            let mut hasher = Sha256::new();
            hasher.update(format!("{}:{}:{}", session_id, start_line, &text[..text.len().min(120)]));
            let uuid = format!("substrate-{}", hex::encode(&hasher.finalize()[..12]));

            sqlx::query(
                r#"INSERT OR REPLACE INTO substrate_nodes
                   (uuid, node_type, text, scope, applies_when,
                    valid_at, invalid_at, expired_at, created_at,
                    source_session_id, source_turn_ref, source_quote, source_actor,
                    confidence, episode_id)
                   VALUES (?,?,?,?,?, ?,NULL,NULL,?, ?,?,?,?, ?,?)"#,
            )
            .bind(&uuid)
            .bind(node_type)
            .bind(text)
            .bind(raw.get("scope").and_then(|v| v.as_str()))
            .bind(raw.get("applies_when").and_then(|v| v.as_str()))
            .bind(&now)
            .bind(&now)
            .bind(&session_id)
            .bind(start_line)
            .bind(raw.get("source").and_then(|v| v.get("quote")).and_then(|v| v.as_str()))
            .bind(raw.get("source").and_then(|v| v.get("actor")).and_then(|v| v.as_str()).unwrap_or("claude"))
            .bind(raw.get("confidence").and_then(|v| v.as_str()).unwrap_or("inferred"))
            .bind(episode_id)
            .execute(sqlite_pool).await
            .map_err(|e| format!("upsert {uuid}: {e}"))?;

            upserted += 1;
        }

        // 6. Emit substrate counter event
        app.emit("substrate:ingested", serde_json::json!({
            "episode_id": episode_id,
            "session_id": session_id,
            "count": upserted,
        })).ok();

        Ok(upserted)
    }

    async fn write_dead_letter(
        pool: &SqlitePool,
        episode_id: &str,
        error_kind: &str,
        raw_output: &str,
    ) -> Result<(), String> {
        let id = uuid::Uuid::new_v4().to_string();
        let truncated = &raw_output[..raw_output.len().min(4096)];
        sqlx::query(
            "INSERT INTO distiller_dead_letters (id, episode_id, error_kind, raw_output, attempt_count)
             VALUES (?, ?, ?, ?, 1)
             ON CONFLICT(id) DO UPDATE SET attempt_count = attempt_count + 1, last_attempt_at = datetime('now')"
        )
        .bind(&id)
        .bind(episode_id)
        .bind(error_kind)
        .bind(truncated)
        .execute(pool).await
        .map_err(|e| format!("dead_letter insert: {e}"))?;
        Ok(())
    }

    async fn write_dead_letter_owned(app: AppHandle, episode_id: String, error_kind: String, raw: Option<String>) {
        // Used for spawn-detached writes where we can't borrow pool reference
        let pool: tauri::State<crate::DbInstances> = app.state();
        if let Ok(guard) = pool.0.read() {
            if let Some(DbPool::Sqlite(sqlite_pool)) = guard.get("sqlite:contract-ide.sqlite") {
                let _ = write_dead_letter(sqlite_pool, &episode_id, &error_kind, raw.as_deref().unwrap_or("")).await;
            }
        }
    }

    /// Subscribe to Phase 10's episode:ingested event. Called from lib.rs setup().
    pub fn init(app: &AppHandle) {
        let app_clone = app.clone();
        app.listen("episode:ingested", move |event| {
            let payload: serde_json::Value = serde_json::from_str(event.payload()).unwrap_or(serde_json::json!({}));
            let Some(episode_id) = payload.get("episode_id").and_then(|v| v.as_str()).map(String::from) else {
                return;
            };
            let app = app_clone.clone();
            tokio::spawn(async move {
                if let Err(e) = distill_episode(&app, &episode_id).await {
                    tracing::warn!("distill_episode({episode_id}) failed: {e}");
                }
            });
        });
    }
    ```

    ```typescript
    import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
    import { getDb } from '../db.js';

    const inputSchema = z.object({
      intent: z.string().min(1, 'intent must be non-empty'),
      limit: z.number().int().positive().max(50).optional().default(5),
    });

    interface SubstrateRow {
      uuid: string;
      text: string;
      applies_when: string | null;
      scope: string | null;
      confidence: string;
      source_session_id: string | null;
      source_turn_ref: number | null;
      source_quote: string | null;

      args: z.infer<typeof inputSchema>,
    ) {

        .prepare(
          `
          SELECT s.uuid, s.text, s.applies_when, s.scope, s.confidence,
                 s.source_session_id, s.source_turn_ref, s.source_quote, s.node_type
          FROM substrate_nodes_fts fts

          WHERE substrate_nodes_fts MATCH ?
            AND s.invalid_at IS NULL
            AND s.node_type = 'constraint'
          ORDER BY fts.rank
          LIMIT ?
          `,
    ```

    ```typescript
    import { z } from 'zod';
    import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
    import { getDb } from '../db.js';

    const inputSchema = z.object({
      scope: z.string().optional(),
      limit: z.number().int().positive().max(50).optional().default(5),
    });

    export async function openQuestions(args: z.infer<typeof inputSchema>) {
      const { scope, limit } = inputSchema.parse(args);
      const db = getDb();

      const rows = scope
        ? (db
            .prepare(
              `SELECT uuid, text, applies_when, scope, confidence, source_session_id, source_turn_ref
               FROM substrate_nodes
               WHERE node_type = 'open_question' AND invalid_at IS NULL AND scope LIKE ?
               ORDER BY valid_at DESC
               LIMIT ?`,
            )
            .all(`${scope}%`, limit))
        : (db
            .prepare(
              `SELECT uuid, text, applies_when, scope, confidence, source_session_id, source_turn_ref
               FROM substrate_nodes
               WHERE node_type = 'open_question' AND invalid_at IS NULL
               ORDER BY valid_at DESC
               LIMIT ?`,
            )
            .all(limit));

      if ((rows as unknown[]).length === 0) {
        const filterDesc = scope ? ` (scope: ${scope})` : '';
        return { content: [{ type: 'text' as const, text: `No open questions found${filterDesc}.` }] };

        .join('\n\n');

      return { content: [{ type: 'text' as const, text }] };
    }

      server.tool(
        'open_questions',
        'List open questions captured by the substrate distiller. Optional scope filter (LIKE-prefix match).',

    }
    ```

    ```rust
    use crate::retrieval::{candidates::candidate_selection, scope::lineage_scope_uuids, SubstrateHit};
    use crate::DbInstances;

    /// Cheap retrieval path for the read-only Substrate side panel.
    /// FTS5 + scope (no LLM rerank). Returns up to `limit` SubstrateHits ordered by FTS5 rank.
    /// If no query is provided, returns all current-truth substrate scoped to the lineage.
    #[tauri::command]
    pub async fn list_substrate_for_atom(
        app: tauri::AppHandle,
        scope_uuid: String,
        query: Option<String>,
        limit: usize,
    ) -> Result<Vec<SubstrateHit>, String> {
        let db: tauri::State<DbInstances> = app.state();
        let guard = db.0.read().map_err(|e| format!("pool lock: {e}"))?;
        let DbPool::Sqlite(pool) = guard.get("sqlite:contract-ide.sqlite").ok_or("no sqlite pool")? else {
            return Err("not sqlite".into());
        };

        let lineage_uuids = lineage_scope_uuids(pool, &scope_uuid).await?;

        // If query provided, FTS5; else, return all current-truth substrate ordered by valid_at DESC
        let query_text = query.unwrap_or_else(|| String::new());
        if !query_text.is_empty() {
            return candidate_selection(pool, &lineage_uuids, &query_text, None, limit).await;
        }

        // No query → return all current-truth substrate (recently captured)
        let nodes: Vec<crate::distiller::types::SubstrateNode> = sqlx::query_as(
            r#"
            SELECT uuid, node_type, text, scope, applies_when,
                   source_session_id, source_turn_ref, source_quote, source_actor,
                   valid_at, invalid_at, expired_at, created_at,
                   confidence, episode_id, invalidated_by
            FROM substrate_nodes
            WHERE invalid_at IS NULL
            ORDER BY valid_at DESC
            LIMIT ?
            "#,
        )
        .bind(limit as i64)
        .fetch_all(pool)
        .await
        .map_err(|e| format!("substrate list: {e}"))?;

        Ok(nodes.into_iter()
            .map(|n| SubstrateHit::from_node(n, crate::retrieval::ScopeUsed::Lineage))
            .collect())
    }
    ```

      onClose: () => void;
    }

    const KIND_ICON: Record<string, string> = {
      constraint: '⚖',
      decision: '✓',
      open_question: '?',
      resolved_question: '✓?',
      attempt: '⚠',
    };

    export function SubstrateSidePanel({ scopeUuid, onClose }: SubstrateSidePanelProps) {
      const [hits, setHits] = useState<SubstrateHit[] | null>(null);
      const [error, setError] = useState<string | null>(null);

      useEffect(() => {

    **Step 2 — Create `contract-ide/src/store/substrate.ts`:**

    ```typescript
    import { create } from 'zustand';
    import { ipcSubstrate } from '../ipc/substrate';

    interface SubstrateStore {
      totalCount: number;
      firstNodeSeen: boolean;
      // Actions
      seedFromIpc: () => Promise<void>;
      onSubstrateIngested: (delta: number) => void;
      markFirstNodeSeen: () => void;
    }

    const STORAGE_KEY = 'substrate.first_node_seen';

    export const useSubstrateStore = create<SubstrateStore>((set, get) => ({
      totalCount: 0,
      firstNodeSeen: typeof window !== 'undefined' && localStorage.getItem(STORAGE_KEY) === 'true',

      seedFromIpc: async () => {
        try {
          const total = await ipcSubstrate.getTotalCount();
          set({ totalCount: total });
          if (total > 0 && !get().firstNodeSeen) {
            // Already past first node before this session — silently mark as seen
            get().markFirstNodeSeen();
          }
        } catch (e) {
          console.warn('seedFromIpc failed', e);
        }
      },

      onSubstrateIngested: (delta: number) => {
        const prev = get().totalCount;
        const next = prev + delta;
        set({ totalCount: next });
        // First-ever transition 0→≥1 → fire toast (Plan 11-05 task wires the toast emit)
        if (prev === 0 && next > 0 && !get().firstNodeSeen) {
          // Emit a custom event so AppShell-level toast handler picks it up
          if (typeof window !== 'undefined') {
            window.dispatchEvent(new CustomEvent('substrate:first-node-toast'));
          }
          get().markFirstNodeSeen();
        }
      },

      markFirstNodeSeen: () => {
        if (typeof window !== 'undefined') {
          localStorage.setItem(STORAGE_KEY, 'true');
        }
        set({ firstNodeSeen: true });
      },
    }));
    ```

    import React, { useEffect } from 'react';
    import { listen } from '@tauri-apps/api/event';
    import { useSubstrateStore } from '../../store/substrate';

    export function SubstrateStatusIndicator() {
      const totalCount = useSubstrateStore((s) => s.totalCount);
      const seedFromIpc = useSubstrateStore((s) => s.seedFromIpc);
      const onSubstrateIngested = useSubstrateStore((s) => s.onSubstrateIngested);

      useEffect(() => {
        // Race-resistant pattern from Phase 5 McpStatusIndicator: seed AND subscribe
        seedFromIpc();
        const unlisten = listen<{ count: number }>('substrate:ingested', (e) => {
          onSubstrateIngested(e.payload.count);
        });
        return () => {
          unlisten.then((u) => u());
        };
      }, [seedFromIpc, onSubstrateIngested]);

      return (
        <div className="flex items-center gap-1.5 text-xs text-muted-foreground" aria-label={`${totalCount} substrate nodes captured`}>
          <span aria-hidden>·</span>
          <span>{totalCount} substrate {totalCount === 1 ? 'node' : 'nodes'} captured</span>
        </div>
      );
    }

   ```rust
   use sha2::{Digest, Sha256};
   use crate::sidecar::frontmatter::{parse_sidecar, ContractFrontmatter};

   #[tauri::command]
   pub fn hash_text(text: String) -> String {
       let mut hasher = Sha256::new();
       hasher.update(text.as_bytes());
       hex::encode(hasher.finalize())
   }

   /// Read and parse an existing sidecar; returns the frontmatter ONLY.
   /// The body is re-entered via the editor buffer on save — we only need
   /// frontmatter to preserve server-derived fields (neighbors, etc.) during
   /// human-pinned saves. Returns None if the sidecar does not exist yet.
   #[tauri::command]
   pub fn read_contract_frontmatter(
       repo_path: String,
       uuid: String,
   ) -> Result<Option<ContractFrontmatter>, String> {
       let path = std::path::Path::new(&repo_path)
           .join(".contracts")
           .join(format!("{uuid}.md"));
       if !path.exists() {
           return Ok(None);
       }
       let raw = std::fs::read_to_string(&path).map_err(|e| e.to_string())?;
       let (fm, _body) = parse_sidecar(&raw).map_err(|e| e.to_string())?;
       Ok(Some(fm))
   }
   ```

1. Rewrite `contract-ide/src/components/inspector/ContractTab.tsx` (Plan 04-01 extracted it as a shell; Task 1 of THIS plan migrated its `onBlur` to the new signature). The component reads `repoPath` from the graph store slice landed in Plan 04-01 Task 1 — DO NOT call `getRepoPath()` IPC here.

   ```tsx
   import { useEffect, useRef } from 'react';
   import { useEditorStore } from '@/store/editor';
   import { useGraphStore } from '@/store/graph';
   import type { ContractNode } from '@/ipc/types';

   const DEBOUNCE_MS = 400;

   export default function ContractTab({ node }: { node: ContractNode | null }) {
     const contractText = useEditorStore((s) => s.contractText);
     const isDirty = useEditorStore((s) => s.isDirty);
     const setContractText = useEditorStore((s) => s.setContractText);
     const saveContract = useEditorStore((s) => s.saveContract);
     const repoPath = useGraphStore((s) => s.repoPath);   // slice added in 04-01 Task 1
     const debounceRef = useRef<number | null>(null);

     // Debounced autosave on typing
     useEffect(() => {
       if (!isDirty || !node || !repoPath) return;
       if (debounceRef.current) window.clearTimeout(debounceRef.current);
       debounceRef.current = window.setTimeout(() => {
         saveContract(repoPath, node).catch(console.error);
       }, DEBOUNCE_MS);
       return () => { if (debounceRef.current) window.clearTimeout(debounceRef.current); };
     }, [contractText, isDirty, node, repoPath, saveContract]);

     // Cmd+S handler — save immediately
     useEffect(() => {
       const onKey = (e: KeyboardEvent) => {
         if ((e.metaKey || e.ctrlKey) && e.key === 's') {
           e.preventDefault();
           if (node && repoPath) saveContract(repoPath, node).catch(console.error);
         }
       };
       document.addEventListener('keydown', onKey);
       return () => document.removeEventListener('keydown', onKey);
     }, [node, repoPath, saveContract]);

     if (!node) {
       return <div className="p-4 text-sm text-muted-foreground">Select a node to edit its contract.</div>;
     }

     return (
       <div className="flex flex-col h-full">
         <div className="px-4 py-2 text-xs text-muted-foreground flex items-center gap-2">
           <span>{isDirty ? 'Editing…' : 'Saved'}</span>
           {node.human_pinned && <span className="px-1.5 py-0.5 rounded bg-amber-500/20 text-amber-300">📌 Pinned</span>}
         </div>
         <textarea
           className="flex-1 w-full resize-none bg-transparent border-0 px-4 py-2 font-sans text-sm leading-relaxed focus:outline-none"
           value={contractText}
           onChange={(e) => setContractText(e.target.value)}
           onBlur={() => {
             if (node && repoPath && isDirty) saveContract(repoPath, node).catch(console.error);
           }}
           placeholder="Describe what this node does — behavior, inputs, outputs…"
           spellCheck={false}
         />
       </div>
     );
   }
   ```

2. Create `contract-ide/src/components/inspector/DriftBadge.tsx`:

   ```tsx
   import { cn } from '@/lib/utils';
   import type { ContractNode } from '@/ipc/types';

   type DriftState = 'synced' | 'drifted' | 'untracked';

   function driftState(node: ContractNode): DriftState {
     if (!node.code_hash || !node.contract_hash) return 'untracked';
     return node.code_hash === node.contract_hash ? 'synced' : 'drifted';
   }

   export default function DriftBadge({ node, onReconcile }: { node: ContractNode | null; onReconcile?: () => void }) {
     if (!node) return null;
     const state = driftState(node);
     if (state === 'untracked') {
       return <span className="text-xs text-muted-foreground">Not derived</span>;
     }
     const isDrift = state === 'drifted';
     return (
       <div className={cn(
         'flex items-center gap-1.5 px-2 py-0.5 rounded text-xs font-medium',
         isDrift ? 'bg-red-500/20 text-red-400' : 'bg-green-500/20 text-green-400'
       )}>
         <span className={cn('w-1.5 h-1.5 rounded-full', isDrift ? 'bg-red-400 animate-pulse' : 'bg-green-400')} />
         {isDrift ? 'Drifted' : 'Synced'}
         {isDrift && (
           <button
             className="ml-2 underline hover:no-underline"
             onClick={onReconcile ?? (() => console.log('[Phase 7] reconcile panel opens here'))}
           >
             Reconcile
           </button>
         )}
       </div>
     );
   }
   ```

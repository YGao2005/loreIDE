```typescript
export interface SessionFile {

  mtime: string;
  lineCount: number;

/** Event payload — `null` fields signal the UI to refetch via getSessionStatus. */
export interface SessionStatusEvent {
  watchingSessions: number | null;
  episodesIngested: number | null;
}

export async function getIngestedSessions(limit = 50): Promise<SessionRow[]> {
  return invoke<SessionRow[]>('get_ingested_sessions', { limit });
}

export async function getBackfillPreview(sessionIds: string[]): Promise<BackfillPreview[]> {
  return invoke<BackfillPreview[]>('get_backfill_preview', { sessionIds });
}

export async function executeBackfill(sessionIds: string[]): Promise<number> {
  return invoke<number>('execute_backfill', { sessionIds });
}

export async function getSessionStatus(): Promise<SessionStatus> {
  return invoke<SessionStatus>('get_session_status', {});
}

export async function listHistoricalSessionFiles(): Promise<SessionFile[]> {
  return invoke<SessionFile[]>('list_historical_session_files', {});
}

export async function subscribeSessionStatus(
  cb: (ev: SessionStatusEvent) => void,
): Promise<UnlistenFn> {
  return listen<SessionStatusEvent>('session:status', (event) => cb(event.payload));
}
```

```typescript
import { useEffect } from 'react';
import {
  getSessionStatus,
  subscribeSessionStatus,
  type SessionStatus,
} from '@/ipc/session';
import { useSessionStore } from '@/store/session';

/**
 * Phase 10 footer indicator: "N sessions · M episodes" + pulse dot.
 *
 * Mirrors McpStatusIndicator: seed from get_session_status on mount, subscribe
 * to session:status events. The seed-from-IPC handles the race where the
 * Rust watcher emits a status event before this component mounts (same
 * mitigation as Plan 05-01 documented in STATE.md).
 *
 * Click target opens the backfill modal (10-04 BackfillModal).
 */
export function SessionStatusIndicator() {
  const status = useSessionStore((s) => s.status);
  const setStatus = useSessionStore((s) => s.setStatus);
  const openBackfillModal = useSessionStore((s) => s.openBackfillModal);

  useEffect(() => {
    let unlisten: (() => void) | undefined;
    let cancelled = false;

    getSessionStatus()
      .then((s) => {
        if (!cancelled) setStatus(s);
      })
      .catch(() => {
        /* ignore — event stream is source of truth */
      });

    subscribeSessionStatus((ev) => {
      // null fields = UI should refetch via getSessionStatus (used by execute_backfill batch path)
      if (ev.watchingSessions === null || ev.episodesIngested === null) {
        getSessionStatus().then((s) => setStatus(s)).catch(() => {});
      } else {
        setStatus({
          watchingSessions: ev.watchingSessions,
          episodesIngested: ev.episodesIngested,
        });
      }
    }).then((fn) => {
      unlisten = fn;
    });

    return () => {
      cancelled = true;
      unlisten?.();
    };
  }, [setStatus]);

  const isWatching = status.watchingSessions > 0;
  const dotClass = isWatching ? 'bg-emerald-500' : 'bg-zinc-400';
  const label = `${status.watchingSessions} session${status.watchingSessions === 1 ? '' : 's'} · ${status.episodesIngested} episode${status.episodesIngested === 1 ? '' : 's'}`;

  return (
    <button
      type="button"
      onClick={openBackfillModal}
      className="flex items-center gap-2 px-3 py-1 text-xs text-muted-foreground hover:text-foreground hover:bg-accent/40 rounded transition-colors"
      title="Click to backfill historical sessions"
      aria-live="polite"
    >
      <span
        className={`h-2 w-2 rounded-full ${dotClass} ${isWatching ? 'animate-pulse' : ''}`}
        aria-hidden
      />
      <span>{label}</span>
    </button>
  );
}
```

```tsx
import { useEffect, useState } from 'react';
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogFooter,
  DialogHeader,
  DialogTitle,
} from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Checkbox } from '@/components/ui/checkbox';
import {
  executeBackfill,
  getBackfillPreview,
  getSessionStatus,
  listHistoricalSessionFiles,
  type BackfillPreview,
  type SessionFile,
} from '@/ipc/session';
import { useSessionStore } from '@/store/session';

/**
 * Backfill flow per Phase 10 SC4:
 * 1. select — list historical session JSONLs in ~/.claude/projects/<cwd-key>/
 *    sorted by mtime DESC (most recent first), default cap at last 30 days.
 * 2. loading-preview (M10) — spinner shown while `get_backfill_preview` IPC
 *    runs filter on each selected session file. Without this state the modal
 *    appears frozen during multi-second IPC calls (filter is sync IO wrapped
 *    in spawn_blocking on the Rust side, but the IPC roundtrip is still
 *    user-visible).
 * 3. preview — run get_backfill_preview against checked sessions, show
 *    estimated tokens + cost per session + total. NO INGESTION YET.
 * 4. confirming — user clicks Confirm; execute_backfill runs; modal closes
 *    on success.
 *
 * Critical UX: SC4 says "nothing ingests automatically without confirmation."
 * The execute_backfill IPC fires only after the user clicks the Confirm
 * button on the preview pane.
 */
type Step = 'select' | 'loading-preview' | 'preview' | 'confirming';

export function BackfillModal() {
  const open = useSessionStore((s) => s.backfillModalOpen);
  const closeModal = useSessionStore((s) => s.closeBackfillModal);

  const [step, setStep] = useState<Step>('select');
  const [files, setFiles] = useState<SessionFile[]>([]);
  const [selected, setSelected] = useState<Set<string>>(new Set());
  const [previews, setPreviews] = useState<BackfillPreview[]>([]);
  const [error, setError] = useState<string | null>(null);
  const [resultEpisodes, setResultEpisodes] = useState<number | null>(null);

  // Reset state on close
  useEffect(() => {
    if (!open) {
      setStep('select');
      setFiles([]);
      setSelected(new Set());
      setPreviews([]);
      setError(null);
      setResultEpisodes(null);
    }
  }, [open]);

  // Fetch historical session list when modal opens (calls 10-03 IPC)
  useEffect(() => {
    if (!open || step !== 'select') return;
    let cancelled = false;
    listHistoricalSessionFiles()
      .then((fs) => {
        if (!cancelled) setFiles(fs);
      })
      .catch((e) => {
        if (!cancelled) setError(typeof e === 'string' ? e : 'Failed to list sessions');
      });
    return () => {
      cancelled = true;
    };
  }, [open, step]);

  function toggle(sessionId: string) {
    setSelected((prev) => {
      const next = new Set(prev);
      if (next.has(sessionId)) {
        next.delete(sessionId);
      } else {
        next.add(sessionId);
      }
      return next;
    });
  }

  async function goPreview() {
    if (selected.size === 0) {
      setError('Select at least one session');
      return;
    }
    setError(null);
    // M10 FIX: transition to loading-preview BEFORE the IPC call so the user
    // sees a spinner instead of a frozen UI during multi-second filter passes.
    setStep('loading-preview');
    try {
      const ids = Array.from(selected);
      const p = await getBackfillPreview(ids);
      setPreviews(p);
      setStep('preview');
    } catch (e) {
      setError(typeof e === 'string' ? e : 'Failed to compute preview');
      setStep('select'); // return to selection so user can retry
    }
  }

  async function confirmExecute() {
    setStep('confirming');
    setError(null);
    try {
      const ids = previews.map((p) => p.sessionId);
      const total = await executeBackfill(ids);
      setResultEpisodes(total);
      // Refresh footer count
      const st = await getSessionStatus();
      useSessionStore.getState().setStatus(st);
    } catch (e) {
      setError(typeof e === 'string' ? e : 'Backfill failed');
      setStep('preview');
    }
  }

  const totalTokens = previews.reduce((sum, p) => sum + p.estimatedTokens, 0);
  const totalCost = previews.reduce((sum, p) => sum + p.estimatedCostUsd, 0);

  return (
    <Dialog open={open} onOpenChange={(o) => (o ? null : closeModal())}>
      <DialogContent className="max-w-2xl">
        <DialogHeader>
          <DialogTitle>Backfill historical sessions</DialogTitle>
          <DialogDescription>
            Ingest past Claude Code sessions from this project into the substrate.
            Phase 10 makes no LLM calls — token estimates are heuristic
            (chars / 4 × Sonnet input rate). Phase 11 distillation runs separately.
          </DialogDescription>
        </DialogHeader>

        {error && (
          <div className="rounded border border-red-500/50 bg-red-500/10 p-3 text-sm text-red-200">
            {error}
          </div>
        )}

        {step === 'select' && (
          <div className="max-h-96 overflow-y-auto">
            {files.length === 0 ? (
              <p className="text-sm text-muted-foreground">
                No historical sessions found for this repo. Run `claude` here to populate.
              </p>
            ) : (
              <table className="w-full text-sm">
                <thead className="sticky top-0 bg-background">
                  <tr className="border-b border-border/40">
                    <th className="w-8" />
                    <th className="text-left">Session</th>
                    <th className="text-right">Lines</th>
                    <th className="text-right">Size</th>
                    <th className="text-left">Date</th>
                  </tr>
                </thead>
                <tbody>
                  {files.map((f) => (
                    <tr key={f.sessionId} className="border-b border-border/20">
                      <td>
                        <Checkbox
                          checked={selected.has(f.sessionId)}
                          onCheckedChange={() => toggle(f.sessionId)}
                        />
                      </td>
                      <td className="font-mono text-xs">{f.sessionId.slice(0, 8)}</td>
                      <td className="text-right">{f.lineCount.toLocaleString()}</td>
                      <td className="text-right">{(f.bytesRaw / 1024).toFixed(0)} KB</td>
                      <td className="text-xs text-muted-foreground">
                        {new Date(f.mtime).toLocaleString()}
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            )}
          </div>
        )}

        {step === 'loading-preview' && (
          <div className="py-12 flex flex-col items-center gap-3 text-muted-foreground" aria-busy="true" aria-live="polite">
            <div
              className="h-6 w-6 rounded-full border-2 border-muted-foreground/30 border-t-muted-foreground animate-spin"
              aria-hidden
            />
            <p className="text-sm">
              Computing preview for {selected.size} session{selected.size === 1 ? '' : 's'}…
            </p>
            <p className="text-xs text-muted-foreground/70">
              Filtering JSONL — no LLM call.
            </p>
          </div>
        )}

        {step === 'preview' && (
          <div className="space-y-4">
            <div className="rounded border border-border/40 bg-muted/20 p-3 text-sm">
              <div className="font-semibold">Total estimate</div>
              <div className="text-muted-foreground">
                ~{totalTokens.toLocaleString()} tokens · ~${totalCost.toFixed(3)}
              </div>
              <div className="mt-1 text-xs text-muted-foreground">
                Heuristic: filtered_chars / 4 × Sonnet input rate ($3 / MTok). No LLM call.
              </div>
            </div>
            <div className="max-h-64 overflow-y-auto">
              <table className="w-full text-sm">
                <thead>
                  <tr className="border-b border-border/40">
                    <th className="text-left">Session</th>
                    <th className="text-right">Episodes</th>
                    <th className="text-right">Tokens</th>
                    <th className="text-right">Cost</th>
                  </tr>
                </thead>
                <tbody>
                  {previews.map((p) => (
                    <tr key={p.sessionId} className="border-b border-border/20">
                      <td className="font-mono text-xs">{p.sessionId.slice(0, 8)}</td>
                      <td className="text-right">{p.episodeCountEstimate}</td>
                      <td className="text-right">{p.estimatedTokens.toLocaleString()}</td>
                      <td className="text-right">${p.estimatedCostUsd.toFixed(4)}</td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          </div>
        )}

        {step === 'confirming' && resultEpisodes === null && (
          <div className="py-8 text-center text-muted-foreground" aria-busy="true" aria-live="polite">
            <div
              className="mx-auto mb-3 h-6 w-6 rounded-full border-2 border-muted-foreground/30 border-t-muted-foreground animate-spin"
              aria-hidden
            />
            Ingesting {previews.length} sessions…
          </div>
        )}

        {step === 'confirming' && resultEpisodes !== null && (
          <div className="rounded border border-emerald-500/50 bg-emerald-500/10 p-4 text-sm">
            Ingested {resultEpisodes} new episode{resultEpisodes === 1 ? '' : 's'} from {previews.length} session{previews.length === 1 ? '' : 's'}.
          </div>
        )}

        <DialogFooter>
          {step === 'select' && (
            <>
              <Button variant="ghost" onClick={closeModal}>
                Cancel
              </Button>
              <Button onClick={goPreview} disabled={selected.size === 0}>
                Preview ({selected.size})
              </Button>
            </>
          )}
          {step === 'loading-preview' && (
            <Button variant="ghost" disabled>
              Computing…
            </Button>
          )}
          {step === 'preview' && (
            <>
              <Button variant="ghost" onClick={() => setStep('select')}>
                Back
              </Button>
              <Button onClick={confirmExecute}>Confirm & Ingest</Button>
            </>
          )}
          {step === 'confirming' && (
            <Button onClick={closeModal} disabled={resultEpisodes === null}>
              {resultEpisodes === null ? 'Ingesting…' : 'Close'}
            </Button>
          )}
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

**Step 1 — SC1 PROGRAMMATIC live latency: starting `claude` produces a sessions row within 2s of first user message (measured to the millisecond, not by stopwatch).**

**M5 FIX — programmatic measurement protocol:**

In TERMINAL A (the measurement terminal), run the following, leaving it ready:

```bash
SQLITE_PATH="$HOME/Library/Application Support/com.contract-ide.app/contract-ide.db"

# Capture the BEFORE timestamp (seconds since epoch + millis) AND the
# matching ISO-8601 cutoff that we'll use to detect new rows.
BEFORE_MS=$(date +%s%3N)
BEFORE_ISO=$(date -u +%Y-%m-%dT%H:%M:%S.000Z)

echo "BEFORE_MS=$BEFORE_MS"
echo "BEFORE_ISO=$BEFORE_ISO"
echo ""
echo "Now switch to TERMINAL B and run: cd /Users/yang/lahacks && claude"
echo "When the assistant responds to your prompt, return here and press Enter."
read
echo "Polling sqlite for new sessions row…"

# Poll sessions table for any row started after BEFORE_ISO. Sub-second poll.
START_POLL_MS=$(date +%s%3N)
TIMEOUT_MS=10000  # absolute timeout 10s — well beyond the 2s SC1 budget
while true; do
  COUNT=$(sqlite3 "$SQLITE_PATH" "SELECT COUNT(*) FROM sessions WHERE started_at > '$BEFORE_ISO'")
  if [ "$COUNT" -gt 0 ]; then
    AFTER_MS=$(date +%s%3N)
    break
  fi
  NOW_MS=$(date +%s%3N)
  if [ $((NOW_MS - START_POLL_MS)) -gt $TIMEOUT_MS ]; then
    AFTER_MS=$NOW_MS
    echo "TIMEOUT — no new session row found within 10s of polling"
    break
  fi
  sleep 0.1
done

# We measure latency as: time from user pressing Enter on the prompt to
# the row appearing. The user's "press Enter when assistant responds" is the
# closest proxy we have to "first user message persisted to disk by claude".
# Note: this measurement INCLUDES the user's reaction time between the
# assistant responding and pressing Enter, which inflates the result by
# ~200-500ms. The TRUE SC1 budget is 2s on the FSEvents path; if total here
# is < 2.5s we pass with margin.
DELTA_MS=$((AFTER_MS - BEFORE_MS))
echo ""
echo "BEFORE_MS=$BEFORE_MS"
echo "AFTER_MS=$AFTER_MS"
echo "DELTA_MS=$DELTA_MS  (target: <= 2500 ms — 2000 budget + 500 reaction)"

sqlite3 "$SQLITE_PATH" "SELECT session_id, started_at, last_seen_at FROM sessions WHERE started_at > '$BEFORE_ISO' ORDER BY started_at ASC"
```
